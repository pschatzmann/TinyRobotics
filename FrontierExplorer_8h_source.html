6/// Strategy for selecting the next frontier cell to explore
7enum class FrontierSelectionStrategy {
8 RANDOM, // Randomly select a frontier cell
9 NEAREST, // Select the frontier cell closest to the robot
10 FARTHEST, // Select the frontier cell farthest from the robot
11 FIRST, // Select the first frontier cell found (e.g., in scan order)
12 LAST, // Select the last frontier cell found
13 SWITCH_FIRST_LAST, // Alternate between first and last on each call
14 CUSTOM // Use a user-defined callback to select the frontier
15};

17/**
18 * @class FrontierExplorer
19 * @brief Generic exploration and frontier-based SLAM utility for grid or
20 * occupancy maps.
21 *
22 * The FrontierExplorer class provides a flexible framework for autonomous
23 * exploration of a 2D or 3D environment using a frontier-based approach. It is
24 * designed to work with any map type that provides cell state access (e.g.,
25 * occupancy grid, bit vector map) and any coordinate type. The class supports
26 * multiple strategies for selecting the next frontier to explore, including
27 * random, nearest, farthest, first, last, and custom selection.
28 *
29 * Typical usage:
30 * - Set the robot's current position with setCurrentPosition().
31 * - Optionally set the frontier selection strategy with setStrategy().
32 * - Optionally provide a custom selection callback for advanced strategies.
33 * - Call getNextFrontier() to retrieve the next cell to explore.
34 *
35 * Features:
36 * - Works with any map type supporting getCell(x, y, state) and
37 * getXCount()/getYCount().
38 * - Finds all frontiers (cells adjacent to unknown space) in the map.
39 * - Supports multiple selection strategies for exploration.
40 * - Allows user-defined custom selection logic.
41 *
42 * @tparam MapType The type of the map (e.g., GridMapBitVector,
43 * GridMap<CellState>)
44 * @tparam Coordinate The coordinate type (default: Coordinate<float>)
45 */

89 /**
90 * @brief Provides the next frontier cell to explore based on the current
91 * strategy.
92 * @param nextCell Output parameter for the selected frontier cell.
93 * @return true if a frontier was found, false otherwise.
94 */
95 bool getNextFrontier(CoordinateT& nextCell) {
96 // find all frontier cells
97 collectFrontiers();
98 record_count_ = frontiers.size();
99 if (frontiers.empty()) return false;
100
101 // switch between FIRST and LAST if the strategy is SWITCH_FIRST_LAST
102 if (switchFirstLast) {
103 // switch between FIRST and LAST on each call
104 strategy_ = (strategy_ == FrontierSelectionStrategy::FIRST)
105 ? FrontierSelectionStrategy::LAST
106 : FrontierSelectionStrategy::FIRST;
107 }
108
109 switch (strategy_) {
110 case FrontierSelectionStrategy::RANDOM:
111 return selectRandom(nextCell);
112 case FrontierSelectionStrategy::NEAREST:
113 return selectNearest(nextCell);
114 case FrontierSelectionStrategy::FARTHEST:
115 return selectFarthest(nextCell);
116 case FrontierSelectionStrategy::FIRST:
117 return selectFirst(nextCell);
118 case FrontierSelectionStrategy::LAST:
119 return selectLast(nextCell);
120 case FrontierSelectionStrategy::CUSTOM:
121 return selectCustom(nextCell);
122 }
123 return false; // Should not reach here
124 }

140 // --- Strategy implementations ---
141 bool selectRandom(CoordinateT& nextCell) {
142 int idx = random(0, frontiers.size());
143 nextCell = frontiers[idx];
144 frontiers.clear();
145 return true;
146 }

148 bool selectNearest(CoordinateT& nextCell) {
149 CoordinateT& result = frontiers[0];
150 float distance = current_pos.distance(result);
151 for (int i = 1; i < frontiers.size(); ++i) {
152 CoordinateT& tmp = frontiers[i];
153 float tmp_distance = current_pos.distance(tmp);
154 if (tmp_distance < distance) {
155 result = tmp;
156 distance = tmp_distance;
157 }
158 }
159 nextCell = result;
160 frontiers.clear();
161 return true;
162 }

164 bool selectFarthest(CoordinateT& nextCell) {
165 CoordinateT& result = frontiers[0];
166 float distance = current_pos.distance(result);
167 for (int i = 1; i < frontiers.size(); ++i) {
168 CoordinateT& tmp = frontiers[i];
169 float tmp_distance = current_pos.distance(tmp);
170 if (tmp_distance > distance) {
171 result = tmp;
172 distance = tmp_distance;
173 }
174 }
175 nextCell = result;
176 frontiers.clear();
177 return true;
178 }

181 nextCell = frontiers.front();
182 frontiers.clear();
183 return true;
184 }

186 bool selectLast(CoordinateT& nextCell) {
187 nextCell = frontiers.back();
188 frontiers.clear();
189 return true;
190 }

193 if (select_callback) {

195 if (idx >= 0 && idx < frontiers.size()) {
196 nextCell = frontiers[idx];

198 return true;
199 }

201 frontiers.clear();
202 return false;

219 /**
220 * @brief Find all frontier cells in the map (cells adjacent to unknown
221 * space).
222 *
223 * A frontier cell is defined as a cell that is FREE (traversable) and has at
224 * least one neighboring cell (in any of the 8 directions) that is UNKNOWN
225 * (unexplored). This method iterates over every cell in the map, and for
226 * each FREE cell, checks all 8 neighbors. If any neighbor is UNKNOWN, the
227 * cell is added to the internal `frontiers` vector. Each cell is considered
228 * only once, so no duplicates occur.
229 *
230 * This is a key step in frontier-based exploration and SLAM, as it
231 * identifies the boundary between explored and unexplored space, guiding the
232 * robot to new areas.
233 *
234 * After calling this method, the `frontiers` vector will contain all current
235 * frontier coordinates, which can then be used by the exploration strategy to
236 * select the next goal.
237 */
238 void collectFrontiers() {
239 frontiers.clear();
240 for (int y = 0; y < map_.getYCount(); ++y) {
241 for (int x = 0; x < map_.getXCount(); ++x) {
242 CellState state;
243 if (!map_.getCell(x, y, state) || state != CellState::FREE) continue;
244 bool found = false;
245 // Check 8 neighbors for UNKNOWN
246 for (int dy = -1; dy <= 1 && !found; ++dy) {
247 for (int dx = -1; dx <= 1 && !found; ++dx) {
248 if (dx == 0 && dy == 0) continue;
249 CellState neighborState;
250 if (map_.getCell(x + dx, y + dy, neighborState) &&
251 neighborState == CellState::UNKNOWN) {
252 // return world coordinates of the frontier cell
253 frontiers.push_back(map_.toWorld(x, y));
254 found = true;
255 }
256 }
257 }
258 }
259 }
