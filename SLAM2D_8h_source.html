9/**
10 * @class SLAM2D
11 * @ingroup slam
12 * @brief 2D Simultaneous Localization and Mapping (SLAM) with frame transforms
13 * and message integration.
14 *
15 * This class implements a modular 2D SLAM system for occupancy grid mapping,
16 * pose estimation, and sensor fusion. It integrates IMU and range sensor data,
17 * uses a frame hierarchy (world, base, lidar) for coordinate transforms, and
18 * supports message-based communication via MessageSource inheritance.
19 *
20 * Features:
21 * - Inherits from MessageSource for message bus integration
22 * - Maintains an occupancy grid map (GridMap<CellState>)
23 * - Integrates IMU2D for pose estimation and publishes IMU state
24 * - Integrates RangeSensor for obstacle detection
25 * - Uses Frame2D and FrameMgr2D for coordinate transforms between frames
26 * - Efficiently updates the map: marks all cells between sensor and obstacle as
27 * FREE, and the obstacle cell as OCCUPIED
28 * - Provides exploration support via getNextFrontier(result) (frontier-based
29 * exploration, returns true if a frontier is found and writes coordinates to
30 * result)
31 * - Supports message subscription/unsubscription for IMU and range sensor
32 *
33 * Typical usage:
34 * @code
35 * Frame2D world, base, lidar;
36 * FrameMgr2D tf;
37 * IMU2D<float> imu;
38 * SLAM2D slam(dx, dy, resolution, world, base, lidar, imu); // Pass IMU2D as parameter
39 * slam.begin();
40 * // Update IMU values (not via SLAM2D, but via the IMU2D instance)
41 * imu.update(accelX, accelY, gyroZ, nowMillis); // IMU update
42 * imu.publish(); // Publish IMU state and update base frame
43 * slam.addRangeMeasurement(distance, angle); // Lidar update
44 * auto& map = slam.getMap();
45 * Coordinate<float> frontier;
46 * if (slam.getNextFrontier(frontier)) {
47 * // Use frontier coordinate
48 * }
49 * @endcode
50 */

112 /**
113 * @brief Find the next frontier cell (boundary between known free and
114 * unknown space).
115 * @return World coordinates of the next frontier cell, or (0,0) if none
116 * found.
117 */
118 bool getNextFrontier(Coordinate<T>& result) const {
119 int xCount = map_.getXCount();
120 int yCount = map_.getYCount();
121 // For each cell, check if it is free and adjacent to unknown
122 for (int cy = 0; cy < yCount; ++cy) {
123 for (int cx = 0; cx < xCount; ++cx) {
124 CellState state;
125 if (!map_.getCell(cx, cy, state)) continue;
126 if (state != CellState::FREE) continue;
127 // Check 4-neighborhood for unknown
128 bool isFrontier = false;
129 for (int dx = -1; dx <= 1 && !isFrontier; ++dx) {
130 for (int dy = -1; dy <= 1 && !isFrontier; ++dy) {
131 if ((dx == 0 && dy == 0) || abs(dx) + abs(dy) != 1)
132 continue; // 4-neighborhood only
133 int nx = cx + dx;
134 int ny = cy + dy;
135 if (nx < 0 || nx >= xCount || ny < 0 || ny >= yCount) continue;
136 CellState neighborState;
137 if (!map_.getCell(nx, ny, neighborState)) continue;
138 if (neighborState == CellState::UNKNOWN) {
139 isFrontier = true;
140 }
141 }
142 }
143 if (isFrontier) {
144 // Convert cell to world coordinates and return
145 result = map_.toWorld(cx, cy);
146 return true;
147 }
148 }
149 }
150 // No frontier found
151 result = Coordinate<T>(0, 0);
152 return false;
153 }

182 /**
183 * @brief Mark all cells between the sensor and the obstacle as FREE, and
184 * the obstacle cell as OCCUPIED.
185 * @param sensor_world World coordinates of the sensor
186 * @param obs_world World coordinates of the obstacle
187 */
188 void markFreeCellsBetweenSensorAndObstacle(const Coordinate<T>& sensor_world,
189 const Coordinate<T>& obs_world) {
190 GridMap<CellState>::Cell cell_obs, cell_sensor;
191 if (map_.worldToCell(obs_world.x, obs_world.y, cell_obs) &&
192 map_.worldToCell(sensor_world.x, sensor_world.y, cell_sensor)) {
193 // Bresenham's line algorithm to mark free cells
194 int x0 = static_cast<int>(cell_sensor.cx);
195 int y0 = static_cast<int>(cell_sensor.cy);
196 int x1 = static_cast<int>(cell_obs.cx);
197 int y1 = static_cast<int>(cell_obs.cy);
198 int dx = abs(x1 - x0), sx = x0 < x1 ? 1 : -1;
199 int dy = -abs(y1 - y0), sy = y0 < y1 ? 1 : -1;
200 int err = dx + dy, e2;
201 int x = x0, y = y0;
202 while (true) {
203 if (x == x1 && y == y1) break; // Stop before marking obstacle cell
204 map_.setCell(x, y, CellState::FREE);
205 e2 = 2 * err;
206 if (e2 >= dy) {
207 err += dy;
208 x += sx;
209 }
210 if (e2 <= dx) {
211 err += dx;
212 y += sy;
213 }
214 }
215 // Mark the obstacle cell as OCCUPIED
216 map_.setCell(x1, y1, CellState::OCCUPIED);
217 }
218 }
