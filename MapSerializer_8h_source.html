9// Reads a line from the stream into 'out'. Returns true if a line was read,
10// false on EOF.
11// Reads a line from the stream into 'out'. Returns true if a line was read,
12// false on EOF.
13static bool readLine(Stream& s, std::string& out) {
14 out.clear();
15 while (true) {
16 int c = s.read();
17 if (c < 0) {
18 // If nothing read, return false to signal EOF
19 return !out.empty();
20 }
21 if (c == '\n') break;
22 out += static_cast<char>(c);
23 }
24 return true;
25}
26/**
27 * @class GridMapSerializer
28 * @brief Utility class for serializing and deserializing grid maps in CSV
29 * format.
30 *
31 * The GridMapSerializer class provides methods to write a grid map to a Print
32 * object (such as Serial) and to read a grid map from a Stream object. It is
33 * designed to work with map types that provide the following interface:
34 * - getXCount(), getYCount(), getResolution(), cellToWorld(), getCell(),
35 * setCell(), resize(), setResolution()
36 *
37 * The serialization format is:
38 * - Header: "GridMap x:<xCount>, y:<yCount>, resolution=<res>"
39 * - One line per cell: "<wx>,<wy>,<value>" (CSV: world x, world y, value)
40 * - If the value is missing, the line is "<wx>,<wy>,"
41 *
42 * This format is compatible with external tools and spreadsheets for easy
43 * inspection and editing.
44 *
45 * Example:
46 * GridMap x:10, y:10, resolution=0.1
47 * 0,0,0
48 * 0,0.1,1
49 * ...
50 *
51 * @tparam Map The map type to serialize/deserialize (e.g., GridMap)
52 * @tparam StateT The cell state type (e.g., float, int, CellState)
53 * @tparam T The numeric type for coordinates (e.g., float)
54 */
55
56template <typename Map, typename StateT = CellState, typename T = float>
57class GridMapSerializer {
58 public:
59 // write map to output
60 size_t write(Map& map, Print& out) {
61 size_t total = 0;
62 char buf[128];
63 int n =
64 snprintf(buf, sizeof(buf), "GridMap x:%d, y:%d, resolution=%g\n",
65 map.getXCount(), map.getYCount(), (double)map.getResolution());
66 if (n > 0) total += out.print(buf);
67 for (int x = 0; x < map.getXCount(); ++x) {
68 for (int y = 0; y < map.getYCount(); ++y) {
69 Coordinate<T> coord;
70 map.cellToWorld(x, y, coord.x, coord.y);
71 StateT state;
72 if (map.getCell(x, y, state)) {
73 int n = snprintf(buf, sizeof(buf), "%g,%g,%g\n", (double)coord.x,
74 (double)coord.y, (double)state);
75 if (n > 0) total += out.print(buf);
76 } else {
77 int n = snprintf(buf, sizeof(buf), "%g,%g,\n", (double)coord.x,
78 (double)coord.y);
79 if (n > 0) total += out.print(buf);
80 }
81 }
82 }
83 out.flush();
84 return total;
85 }
86
87 // read map from input
88 size_t readFrom(Map& map, Stream& in) {
89 size_t total = 0;
90 int x, y;
91 float res;
92
93 // Read header
94 std::string header;
95 if (!readLine(in, header)) return 0;
96 total += header.length() + 1;
97 if (sscanf(header.c_str(), "GridMap x:%d, y:%d, resolution=%f", &x, &y,
98 &res) != 3)
99 return 0;
100 if (x <= 0 || y <= 0 || res <= 0) return 0;
101 map.resize(x, y);
102 map.setResolution(res);
103
104 // Read cell values
105 int cellCount = x * y;
106 int filled = 0;
107 while (filled < cellCount) {
108 std::string line;
109 if (!readLine(in, line)) return 0;
110 total += line.length() + 1;
111 float wx, wy, value;
112 int parsed = sscanf(line.c_str(), "%f,%f,%f", &wx, &wy, &value);
113 if (parsed < 2) return 0; // must have at least x and y
114 Coordinate<T> coord(static_cast<T>(wx), static_cast<T>(wy));
115 if (parsed == 3) {
116 map.setCell(coord, (StateT)value);
117 }
118 ++filled;
119 }
120 return total;
121 }
122};

124/**
125 * @class PathMapSerializer
126 * @brief Utility class for serializing and deserializing path maps (sequences
127 * of segments) in CSV format.
128 *
129 * The PathMapSerializer class provides methods to write a path map (a
130 * sequence of segments, each with a 'from' and 'to' coordinate, a cost, and a
131 * directed flag) to a Print object (such as Serial) and to read a path map
132 * from a Stream object. It is designed to work with map types that provide:
133 * - size(), operator[](size_t), and addSegment(from, to, cost, directed)
134 * - Each segment must have 'from' and 'to' members of type CoordinateT,
135 * plus 'cost' (double/float) and 'directed' (bool)
136 *
137 * The serialization format is:
138 * - One line per segment:
139 * "<from.x>,<from.y>,<to.x>,<to.y>,<cost>,<directed>" (CSV)
140 * - <cost> is a floating-point value, <directed> is 0 (false) or 1 (true)
141 * - If <cost> and <directed> are missing, defaults are used (cost=1.0,
142 * directed=false)
143 *
144 * This format is compatible with external tools and spreadsheets for easy
145 * inspection and editing.
146 *
147 * Example:
148 * 0,0,1,1,2.5,1
149 * 1,1,2,2,1.0,0
150 * ...
151 *
152 * @tparam MapT The map type to serialize/deserialize (must support segment
153 * access and addSegment)
154 * @tparam CoordinateT The coordinate type for segment endpoints (e.g.,
155 * Coordinate<float>)
156 */
157
158template <typename MapT, typename CoordinateT = Coordinate<float>>
159class PathMapSerializer {
160 using T = typename CoordinateT::value_type;
161
162 size_t write(MapT& map, Print& out) {
163 char buf[160];
164 size_t total = 0;
165 size_t len = map.size();
166 for (size_t i = 0; i < len; ++i) {
167 auto& segment = map[i];
168 auto& from = segment.from;
169 auto& to = segment.to;
170 double cost = segment.cost;
171 bool directed = segment.directed;
172 int n = snprintf(buf, sizeof(buf), "%g,%g,%g,%g,%g,%d\n", (double)from.x,
173 (double)from.y, (double)to.x, (double)to.y, cost,
174 directed ? 1 : 0);
175 if (n > 0) total += out.print(buf);
176 }
177 out.flush();
178 return total;
179 }
180
181 size_t readFrom(MapT& map, Stream& in) {
182 size_t total = 0;
183 std::string line;
184 while (readLine(in, line)) {
185 total += line.length() + 1;
186 float fx, fy, tx, ty, cost = 1.0f;
187 int directedInt = 0;
188 int parsed = sscanf(line.c_str(), "%f,%f,%f,%f,%f,%d", &fx, &fy, &tx, &ty,
189 &cost, &directedInt);
190 if (parsed < 4) return 0; // must have at least coordinates
191 CoordinateT from(static_cast<T>(fx), static_cast<T>(fy));
192 CoordinateT to(static_cast<T>(tx), static_cast<T>(ty));
193 double segCost = (parsed >= 5) ? cost : 1.0;
194 bool directed = (parsed == 6) ? (directedInt != 0) : false;
195 PathSegment<CoordinateT> segment{from, to, segCost, directed};
196 map.addSegment(segment);
197 }
198 return total;
199 }
200};

202/**
203 * @class PointCloudSerializer
204 * @brief Utility class for serializing and deserializing PointCloud objects,
205 * including points, voxels, and attributes.
206 *
207 * The PointCloudSerializer class provides methods to write a PointCloud to a
208 * Print object (such as Serial) and to read a PointCloud from a Stream object.
209 * The format includes a header with attributes, points, and voxels.
210 *
211 * Format:
212 * #PointCloud voxelSize:<float> is3d:<0|1>
213 * bounds:<minx>,<miny>,<minz>,<maxx>,<maxy>,<maxz> #Points x1,y1,z1
214 * ...
215 * #Voxels
216 * vx1,vy1,vz1
217 * ...
218 *
219 * @tparam PointCloudT The PointCloud type to serialize/deserialize (e.g.,
220 * PointCloud<float>)
221 * @tparam T The numeric type for coordinates (e.g., float)
222 */
223template <typename PointCloudT>
224class PointCloudSerializer {
225 using T = typename PointCloudT::value_type;
226
227 public:
228 // Write all attributes, points, and voxels
229 size_t write(const PointCloudT& cloud, Print& out) {
230 size_t total = 0;
231 char buf[128];
232 // Header
233 const auto& b = cloud.bounds();
234 sprintf(buf, "#PointCloud voxelSize:%g is3d:%d bounds:%g,%g,%g,%g,%g,%g\n",
235 (double)cloud.voxelSize(), cloud.is3D() ? 1 : 0, (double)b.min.x,
236 (double)b.min.y, (double)b.min.z, (double)b.max.x, (double)b.max.y,
237 (double)b.max.z);
238 total += out.print(buf);
239 // Points
240 total += out.print("#Points\n");
241 for (const auto& pt : cloud.points()) {
242 sprintf(buf, "%g,%g,%g\n", (double)pt.x, (double)pt.y, (double)pt.z);
243 total += out.print(buf);
244 }
245 // Voxels
246 total += out.print("#Voxels\n");
247 for (auto it = cloud.beginVoxels(); it != cloud.endVoxels(); ++it) {
248 sprintf(buf, "%d,%d,%d\n", it->x, it->y, it->z);
249 total += out.print(buf);
250 }
251 out.flush();
252 return total;
253 }
254
255 // Read all attributes, points, and voxels
256 size_t readFrom(PointCloudT& cloud, Stream& in) {
257 size_t total = 0;
258 std::string line;
259 // Header
260 if (!readLine(in, line)) return 0;
261 total += line.length() + 1;
262 float voxelSize = 0.0f, minx, miny, minz, maxx, maxy, maxz;
263 int is3d = 0;
264 if (sscanf(line.c_str(),
265 "#PointCloud voxelSize:%f is3d:%d bounds:%f,%f,%f,%f,%f,%f",
266 &voxelSize, &is3d, &minx, &miny, &minz, &maxx, &maxy,
267 &maxz) != 8)
268 return 0;
269 cloud.set3D(is3d != 0);
270 cloud.setBounds(minx, miny, minz, maxx, maxy, maxz);
271 cloud.buildVoxelGrid(voxelSize);
272 // Expect #Points
273 if (!readLine(in, line)) return 0;
274 total += line.length() + 1;
275 if (line.find("#Points") != 0) return 0;
276 // Read points
277 while (readLine(in, line)) {
278 total += line.length() + 1;
279 if (line.find("#Voxels") == 0) break;
280 float x, y, z = 0.0f;
281 int parsed = sscanf(line.c_str(), "%f,%f,%f", &x, &y, &z);
282 if (parsed < 2) return 0;
283 cloud.add(static_cast<T>(x), static_cast<T>(y), static_cast<T>(z));
284 }
285 // Read voxels
286 while (readLine(in, line)) {
287 total += line.length() + 1;
288 int vx, vy, vz;
289 if (sscanf(line.c_str(), "%d,%d,%d", &vx, &vy, &vz) == 3) {
290 cloud.addVoxel((float)vx * voxelSize, (float)vy * voxelSize,
291 (float)vz * voxelSize);
292 }
293 }
294 return total;
295 }
296};
