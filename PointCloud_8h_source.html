30/**
31 * @class PointCloud
32 * @ingroup maps
33 * @brief Represents a 3D point cloud for robotics and mapping applications.
34 *
35 * The PointCloud class manages a collection of 3D points and provides:
36 * - Efficient addition of points (individually, from arrays, or from
37 * Coordinate).
38 * - Automatic bounding box tracking for fast spatial queries.
39 * - Voxel grid downsampling to reduce point density while preserving structure.
40 * - Fast occupancy queries via a voxel grid (for collision checking or
41 * mapping).
42 * - Optional "live" voxel grid mode for real-time occupancy updates as points
43 * are added.
44 *
45 * Usage scenarios include 3D mapping, obstacle detection, SLAM, and environment
46 * representation. The class is designed for embedded and robotics use,
47 * balancing efficiency and flexibility.
48 *
49 * Key features:
50 * - Add, clear, and access points.
51 * - Retrieve the bounding box (min/max x, y, z).
52 * - Downsample using a voxel grid (keep one point per voxel).
53 * - Build and query a voxel occupancy grid for fast lookups.
54 * - Optionally enable live voxel grid updates for real-time occupancy.
55 *
56 * Example:
57 * @code
58 * PointCloud cloud;
59 * cloud.add(1.0, 2.0, 0.5);
60 * cloud.buildVoxelGrid(0.1f); // 10cm voxels
61 * bool occ = cloud.isOccupied(1.0, 2.0, 0.5);
62 * @endcode
63 */

72 struct Key {

74 bool operator==(const Key& other) const {
75 return x == other.x && y == other.y && z == other.z;

79 struct KeyHash {
80 size_t operator()(const Key& k) const {
81 return ((std::hash<int>()(k.x) ^ (std::hash<int>()(k.y) << 1)) >> 1) ^
82 (std::hash<int>()(k.z) << 1);

133 /// Simple voxel downsampling: e.g. 0.5 m to reduce number of points while
134 /// keeping overall structure.
135 PointCloud voxelDownsample(float voxelSize) const {
136 PointCloud out;
137 if (voxelSize <= 0.0f) return out;
138
139 std::unordered_map<Key, Point3D<T>, KeyHash> voxels;
140
141 for (const auto& p : points_) {
142 Key key{int(std::floor(p.x / voxelSize)),
143 int(std::floor(p.y / voxelSize)),
144 int(std::floor(p.z / voxelSize))};
145
146 // Keep first point per voxel (simple version)
147 if (voxels.find(key) == voxels.end()) voxels[key] = p;
148 }
149
150 for (const auto& kv : voxels)
151 out.add(kv.second.x, kv.second.y, kv.second.z);
152
153 return out;
154 }

192 /**
193 * @brief Returns the free (unoccupied) neighboring coordinates of a given
194 * point in the voxel grid.
195 *
196 * If the voxel grid is built (voxelSize_ > 0), this method returns all
197 * adjacent unoccupied voxels (6-connectivity in 3D, or 4/8-connectivity in 2D
198 * if z=0). Only neighbors that are not present in the voxel grid are
199 * returned.
200 *
201 * @param from The coordinate to find neighbors for.
202 * @return std::vector<Coordinate<T>> List of free neighboring
203 * coordinates.
204 */
205 std::vector<Coordinate<T>> getNeighbors(Coordinate<T> from) const override {
206 std::vector<Coordinate<T>> neighbors;
207 if (voxelSize_ <= 0.0f) return neighbors;
208 TRLogger.debug("-------------------");
209 TRLogger.debug(
210 "Finding neighbors for: %s with bounding box min:(%f,%f,%f) / "
211 "max:(%f,%f,%f)",
212 from.toCString(), bounds_.min.x, bounds_.min.y, bounds_.min.z,
213 bounds_.max.x, bounds_.max.y, bounds_.max.z);
214
215 // Compute voxel index of 'from'
216 int x0 = int(std::floor(from.x / voxelSize_));
217 int y0 = int(std::floor(from.y / voxelSize_));
218 int z0 = int(std::floor(from.z / voxelSize_));
219
220 if (is_3d) {
221 // 6-connectivity in 3D (face neighbors)
222 const int dx[6] = {1, -1, 0, 0, 0, 0};
223 const int dy[6] = {0, 0, 1, -1, 0, 0};
224 const int dz[6] = {0, 0, 0, 0, 1, -1};
225 for (int i = 0; i < 6; ++i) {
226 int nx = x0 + dx[i];
227 int ny = y0 + dy[i];
228 int nz = z0 + dz[i];
229 // Convert voxel index back to coordinate (center of voxel)
230 T cx = (nx + 0.5f) * voxelSize_;
231 T cy = (ny + 0.5f) * voxelSize_;
232 T cz = (nz + 0.5f) * voxelSize_;
233 // Check bounding box
234 if (cx < bounds_.min.x || cx > bounds_.max.x || cy < bounds_.min.y ||
235 cy > bounds_.max.y || cz < bounds_.min.z || cz > bounds_.max.z) {
236 continue;
237 }
238 Key nkey{nx, ny, nz};
239 // Only add if NOT occupied (free neighbor)
240 if (voxelGrid_.find(nkey) == voxelGrid_.end()) {
241 neighbors.emplace_back(cx, cy, cz);
242 }
243 }
244 } else {
245 // 4-connectivity in 2D (z=0 plane)
246 const int dx[4] = {1, -1, 0, 0};
247 const int dy[4] = {0, 0, 1, -1};
248 for (int i = 0; i < 4; ++i) {
249 int nx = x0 + dx[i];
250 int ny = y0 + dy[i];
251 int nz = z0; // keep z the same
252 T cx = (nx + 0.5f) * voxelSize_;
253 T cy = (ny + 0.5f) * voxelSize_;
254 T cz = 0.0f; // Always use z=0 for 2D
255 // Check bounding box (z=0 plane)
256 if (cx < bounds_.min.x || cx > bounds_.max.x || cy < bounds_.min.y ||
257 cy > bounds_.max.y) {
258 continue;
259 }
260 Key nkey{nx, ny, nz};
261 if (voxelGrid_.find(nkey) == voxelGrid_.end()) {
262 neighbors.emplace_back(cx, cy, cz);
263 }
264 }
265 }
266 for (auto& n : neighbors) {
267 TRLogger.debug("Neighbor: %s", n.toCString());
268 }
269 TRLogger.debug("-------------------");
270
271 return neighbors;
272 }
