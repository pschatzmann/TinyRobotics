28/**
29 * @class PointCloud
30 * @ingroup maps
31 * @brief Represents a 3D point cloud for robotics and mapping applications.
32 *
33 * The PointCloud class manages a collection of 3D points and provides:
34 * - Efficient addition of points (individually, from arrays, or from
35 * Coordinate).
36 * - Automatic bounding box tracking for fast spatial queries.
37 * - Voxel grid downsampling to reduce point density while preserving structure.
38 * - Fast occupancy queries via a voxel grid (for collision checking or
39 * mapping).
40 * - Optional "live" voxel grid mode for real-time occupancy updates as points
41 * are added.
42 *
43 * Usage scenarios include 3D mapping, obstacle detection, SLAM, and environment
44 * representation. The class is designed for embedded and robotics use,
45 * balancing efficiency and flexibility.
46 *
47 * Key features:
48 * - Add, clear, and access points.
49 * - Retrieve the bounding box (min/max x, y, z).
50 * - Downsample using a voxel grid (keep one point per voxel).
51 * - Build and query a voxel occupancy grid for fast lookups.
52 * - Optionally enable live voxel grid updates for real-time occupancy.
53 *
54 * Example:
55 * @code
56 * PointCloud cloud;
57 * cloud.add(1.0, 2.0, 0.5);
58 * cloud.buildVoxelGrid(0.1f); // 10cm voxels
59 * bool occ = cloud.isOccupied(1.0, 2.0, 0.5);
60 * @endcode
61 */

71 bool operator==(const Key& other) const {

76 struct KeyHash {
77 size_t operator()(const Key& k) const {
78 return ((std::hash<int>()(k.x) ^ (std::hash<int>()(k.y) << 1)) >> 1) ^
79 (std::hash<int>()(k.z) << 1);

138 /// Simple voxel downsampling: e.g. 0.5 m to reduce number of points while
139 /// keeping overall structure.
140 PointCloud voxelDownsample(float voxelSize) const {
141 PointCloud out;
142 if (voxelSize <= 0.0f) return out;
143
144 std::unordered_map<Key, Point3D, KeyHash> voxels;
145
146 for (const auto& p : points_) {
147 Key key{int(std::floor(p.x / voxelSize)),
148 int(std::floor(p.y / voxelSize)),
149 int(std::floor(p.z / voxelSize))};
150
151 // Keep first point per voxel (simple version)
152 if (voxels.find(key) == voxels.end()) voxels[key] = p;
153 }
154
155 for (const auto& kv : voxels)
156 out.add(kv.second.x, kv.second.y, kv.second.z);
157
158 return out;
159 }

161 /// Builds a voxel grid for fast occupancy queries: A voxel is a grid cell
162 void buildVoxelGrid(float voxelSize) {
163 voxelGrid_.clear();
164 voxelSize_ = voxelSize;

166 for (const auto& p : points_) {
167 addVoxel(p.x, p.y, p.z);

172 TRLogger.debug("Voxel: (%d, %d, %d)", key.x, key.y, key.z);
173 }
174 }

197 /**
198 * @brief Returns the free (unoccupied) neighboring coordinates of a given
199 * point in the voxel grid.
200 *
201 * If the voxel grid is built (voxelSize_ > 0), this method returns all
202 * adjacent unoccupied voxels (6-connectivity in 3D, or 4/8-connectivity in 2D
203 * if z=0). Only neighbors that are not present in the voxel grid are
204 * returned.
205 *
206 * @param from The coordinate to find neighbors for.
207 * @return std::vector<Coordinate<DistanceM>> List of free neighboring
208 * coordinates.
209 */
210 std::vector<Coordinate<DistanceM>> getNeighbors(
211 const Coordinate<DistanceM>& from) const {
212 std::vector<Coordinate<DistanceM>> neighbors;
213 if (voxelSize_ <= 0.0f) return neighbors;
214 TRLogger.debug("-------------------");
215 TRLogger.debug(
216 "Finding neighbors for: %s with bounding box min:(%f,%f,%f) / "
217 "max:(%f,%f,%f)",
218 from.toCString(), bounds_.min.x, bounds_.min.y, bounds_.min.z,
219 bounds_.max.x, bounds_.max.y, bounds_.max.z);
220
221 // Compute voxel index of 'from'
222 int x0 = int(std::floor(from.x / voxelSize_));
223 int y0 = int(std::floor(from.y / voxelSize_));
224 int z0 = int(std::floor(from.z / voxelSize_));
225
226 if (is_3d) {
227 // 6-connectivity in 3D (face neighbors)
228 const int dx[6] = {1, -1, 0, 0, 0, 0};
229 const int dy[6] = {0, 0, 1, -1, 0, 0};
230 const int dz[6] = {0, 0, 0, 0, 1, -1};
231 for (int i = 0; i < 6; ++i) {
232 int nx = x0 + dx[i];
233 int ny = y0 + dy[i];
234 int nz = z0 + dz[i];
235 // Convert voxel index back to coordinate (center of voxel)
236 DistanceM cx = (nx + 0.5f) * voxelSize_;
237 DistanceM cy = (ny + 0.5f) * voxelSize_;
238 DistanceM cz = (nz + 0.5f) * voxelSize_;
239 // Check bounding box
240 if (cx < bounds_.min.x || cx > bounds_.max.x || cy < bounds_.min.y ||
241 cy > bounds_.max.y || cz < bounds_.min.z || cz > bounds_.max.z) {
242 continue;
243 }
244 Key nkey{nx, ny, nz};
245 // Only add if NOT occupied (free neighbor)
246 if (voxelGrid_.find(nkey) == voxelGrid_.end()) {
247 neighbors.emplace_back(cx, cy, cz);
248 }
249 }
250 } else {
251 // 4-connectivity in 2D (z=0 plane)
252 const int dx[4] = {1, -1, 0, 0};
253 const int dy[4] = {0, 0, 1, -1};
254 for (int i = 0; i < 4; ++i) {
255 int nx = x0 + dx[i];
256 int ny = y0 + dy[i];
257 int nz = z0; // keep z the same
258 DistanceM cx = (nx + 0.5f) * voxelSize_;
259 DistanceM cy = (ny + 0.5f) * voxelSize_;
260 DistanceM cz = 0.0f; // Always use z=0 for 2D
261 // Check bounding box (z=0 plane)
262 if (cx < bounds_.min.x || cx > bounds_.max.x || cy < bounds_.min.y ||
263 cy > bounds_.max.y) {
264 continue;
265 }
266 Key nkey{nx, ny, nz};
267 if (voxelGrid_.find(nkey) == voxelGrid_.end()) {
268 neighbors.emplace_back(cx, cy, cz);
269 }
270 }
271 }
272 for (auto& n : neighbors) {
273 TRLogger.debug("Neighbor: %s", n.toCString());
274 }
275 TRLogger.debug("-------------------");
276
277 return neighbors;
278 }
