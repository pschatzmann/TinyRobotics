100 bool sendOdometry(const MotionState3D state) {
101 nav_msgs__msg__Odometry msg;
102 // Zero/init all fields
103 memset(&msg, 0, sizeof(msg));
104
105 // Position
106 msg.pose.pose.position.x = state.getPosition().x;
107 msg.pose.pose.position.y = state.getPosition().y;
108 msg.pose.pose.position.z = state.getPosition().z;
109
110 // Orientation (convert yaw/pitch/roll to quaternion)
111 float cy = cosf(state.getOrientation().yaw * 0.5f);
112 float sy = sinf(state.getOrientation().yaw * 0.5f);
113 float cp = cosf(state.getOrientation().pitch * 0.5f);
114 float sp = sinf(state.getOrientation().pitch * 0.5f);
115 float cr = cosf(state.getOrientation().roll * 0.5f);
116 float sr = sinf(state.getOrientation().roll * 0.5f);
117 msg.pose.pose.orientation.w = cr * cp * cy + sr * sp * sy;
118 msg.pose.pose.orientation.x = sr * cp * cy - cr * sp * sy;
119 msg.pose.pose.orientation.y = cr * sp * cy + sr * cp * sy;
120 msg.pose.pose.orientation.z = cr * cp * sy - sr * sp * cy;
121
122 // Linear velocity
123 msg.twist.twist.linear.x = state.getSpeed().x;
124 msg.twist.twist.linear.y = state.getSpeed().y;
125 msg.twist.twist.linear.z = state.getSpeed().z;
126
127 // Angular velocity
128 msg.twist.twist.angular.x = state.getAngularVelocity().x;
129 msg.twist.twist.angular.y = state.getAngularVelocity().y;
130 msg.twist.twist.angular.z = state.getAngularVelocity().z;
131
132 // Optionally: set frame_id, child_frame_id, covariance, etc.
133 // strncpy(msg.header.frame_id, "odom", sizeof(msg.header.frame_id));
134 // strncpy(msg.child_frame_id, "base_link", sizeof(msg.child_frame_id));
135
136 return sendOdometry(msg);
137 }
