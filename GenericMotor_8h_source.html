7/**
8 * @class GenericMotor
9 * @ingroup motors
10 * @brief Motor abstraction for integrating external/custom motor drivers using
11 * callbacks.
12 *
13 * This class provides a flexible interface for controlling motors that do not
14 * fit the standard interface. The user supplies an external motor object (as a
15 * void pointer) and callback functions for starting/stopping the motor and for
16 * setting the speed or angle. This allows integration of custom motor drivers
17 * or hardware with the TinyRobotics framework.
18 *
19 * - The @c valueCB callback is mandatory and is called with the desired speed
20 * or angle value.
21 * - The @c beginCB and @c endCB callbacks are optional and can be used to
22 * define custom start/stop logic.
23 * - The user motor object can be accessed via @c getMotor<T>().
24 *
25 * Example usage:
26 * @code
27 * MyMotorDriver driver;
28 * GenericMotor motor(0, &driver);
29 * motor.setValueCallback([](int8_t value, GenericMotor& m) {
30 * MyMotorDriver* drv = m.getMotor<MyMotorDriver>();
31 * drv->setPWM(value);
32 * });
33 * motor.setBeginCallback([](GenericMotor& m) {
34 * // Custom start logic
35 * return true;
36 * });
37 * motor.setEndCallback([](GenericMotor& m) {
38 * // Custom stop logic
39 * });
40 * @endcode
41 */

43template <typename T = float>

52 bool begin() override {
53 if (valueCB == nullptr) {
54 return false; // Callbacks not defined, cannot start
55 }
56 bool result = true;
57 // Call optional callback to start the motor
58 if (beginCB) result = beginCB(*this);
59 return result;
60 }
61
62 // Set value as percentage (-100 to 100)
63 bool setValuePercent(T percent) override {
64 value = constrain(percent, -100.0f, 100.0f);
65 if (valueCB) {
66 valueCB(percent, *this);
67 }
68 return true;
69 }
70
71 T getValuePercent() const override { return value; }
72
73 void setAngle(T angle) {
74 float valuePercent =
75 map(angle, -90.0f, 90.0f, -100.0f, 100.0f); // Map angle to percentage
76 setValuePercent(
77 valuePercent); // For simplicity, treat angle as a percentage value
78 }

80 T getAngle() {
81 return map(value, -100.0f, 100.0f, -90.0f,
82 90.0f); // Map percentage back to angle
83 }

85 void end() override {

87 endCB(*this); // Call callback to stop the motor
88 }
89 }
