53/**
54 * @brief Represents a graph of valid path segments (edges) between coordinates
55 * (nodes).
56 *
57 * PathMap is used for pathfinding algorithms (A*, Dijkstra, etc.) to find
58 * optimal paths between nodes. The map can contain both directed (one-way) and
59 * undirected (two-way) segments.
60 *
61 * Features:
62 * - Add segments (edges) between coordinates (nodes)
63 * - Query outgoing and incoming segments for a given node
64 * - Supports both directed and undirected graphs
65 *
66 * Typical usage: model a road network, robot navigation graph, or any scenario
67 * where movement is constrained to a set of valid connections.
68 *
69 * @tparam Coordinate The coordinate type (e.g., 2D or 3D point).
70 */
71template <typename CoordinateT = Coordinate<float>>
72class PathMap : public IMapNeighbors<typename std::remove_reference<decltype(std::declval<CoordinateT>().x)>::type> {
73 public:
74 PathMap() = default;
75
76 void addSegment(const CoordinateT& from, const CoordinateT& to,
77 bool directed = false) {
78 auto distance = from.distance(to); // Example cost based on distance
79 PathSegment<CoordinateT> segment{from, to, distance,
80 directed}; // Default cost = 1.0
81 segments.push_back(segment);
82 }
83
84 std::vector<PathSegment<CoordinateT>> getSegments(
85 const CoordinateT& from) const {
86 std::vector<PathSegment<CoordinateT>> result;
87 for (const auto& seg : segments) {
88 if (seg.from == from) {
89 result.push_back(seg);
90 }
91 }
92 for (const auto& seg : segments) {
93 if (seg.to == from && !seg.directed) {
94 PathSegment<CoordinateT> reverse_seg = seg;
95 std::swap(reverse_seg.from, reverse_seg.to);
96 result.push_back(reverse_seg);
97 }
98 }
99 return result;
100 }
101
102 // IMap interface: getNeighbors
103 std::vector<CoordinateT> getNeighbors(CoordinateT from) const override {
104 std::vector<CoordinateT> neighbors;
105 for (const auto& seg : segments) {
106 if (seg.from == from) {
107 neighbors.push_back(seg.to);
108 }
109 if (seg.to == from && !seg.directed) {
110 neighbors.push_back(seg.from);
111 }
112 }
113 return neighbors;
114 }
115
116 // IMap interface: isValid (always true for graph nodes in PathMap)
117 bool isValid(const Coordinate<float>& coord) const override {
118 // A node is valid if it appears in any segment
119 for (const auto& seg : segments) {
120 if (seg.from == coord || seg.to == coord) return true;
121 }
122 return false;
123 }
124
125 protected:
126 std::vector<PathSegment<CoordinateT>,
127 AllocatorPSRAM<PathSegment<CoordinateT>>>
128 segments;
129};
