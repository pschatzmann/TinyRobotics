17/**
18 * @class Quadrotor
19 * @ingroup vehicles
20 * @brief Simple quadrotor (quadcopter) model with 4-motor control.
21 *
22 * This class abstracts the control of a basic quadcopter:
23 * - 4 motors (front left, front right, rear left, rear right) via HBridge
24 * - Throttle, roll, pitch, and yaw control
25 *
26 * Usage Example (with setPins):
27 * @code
28 * Quadrotor quad;
29 * quad.setPins(FRONT_LEFT, m1_in1, m1_in2); // front left
30 * quad.setPins(FRONT_RIGHT, m2_in1, m2_in2); // front right
31 * quad.setPins(REAR_LEFT, m3_in1, m3_in2); // rear left
32 * quad.setPins(REAR_RIGHT, m4_in1, m4_in2); // rear right
33 * quad.setThrottle(60); // 60% throttle
34 * quad.setRoll(10); // roll right
35 * quad.setPitch(-5); // pitch down
36 * quad.setYaw(15); // yaw right
37 * @endcode
38 */
39
40template <typename MotorT = BrushedMotor<float>>
41class Quadrotor : public Vehicle {

45 /**
46 * @brief Set the pins for a specific motor (0=front left, 1=front right,

49 void setPins(QuadrotorMotorNo motor, int in1, int in2) {

54 /** Set throttle (0-100%) for all motors */
55 void setThrottle(float percent) {
56 throttle_ = constrain(percent, 0.0f, 100.0f);
57 updateMotors();
58 }
59
60 /** Set roll (-100 to 100): positive = right, negative = left */
61 void setRoll(float percent) {
62 roll_ = constrain(percent, -100.0f, 100.0f);
63 updateMotors();
64 }
65
66 /** Set pitch (-100 to 100): positive = forward, negative = backward */
67 void setPitch(float percent) {
68 pitch_ = constrain(percent, -100.0f, 100.0f);
69 updateMotors();
70 }
71
72 /** Set yaw (-100 to 100): positive = right, negative = left */
73 void setYaw(float percent) {
74 yaw_ = constrain(percent, -100.0f, 100.0f);
75 updateMotors();
76 }

78 /** Stop all motors and reset state */
79 void end() {
80 for (int i = 0; i < 4; ++i) {
81 motors_[i].end();
82 }
83 throttle_ = 0.0f;
84 roll_ = 0.0f;
85 pitch_ = 0.0f;
86 yaw_ = 0.0f;
87 }

89 /**
90 * @brief Set a calibration gain for a specific motor (default 1.0).
91 * @param motor Motor index (QuadrotorMotorNo)
92 * @param gain Gain factor (e.g., 1.05 for +5% output)

95 motorGain_[motor] = gain;
96 }

98 bool isPinsSet() const {
99 for (int i = 0; i < 4; ++i) {
100 if (!motors_[i].isPinsSet()) return false;
101 }
102 return true;
103 }

105 bool onMessage(const Message<float>& msg) override {
106 float angle;
107 if (msg.origin != MessageOrigin::RemoteControl)
108 return false; // Only handle RC messages
109 switch (msg.content) {
110 case MessageContent::Throttle:
111 if (msg.unit != Unit::Percent) return false;
112 setThrottle(static_cast<int>(msg.value));
113 return true;
114 case MessageContent::Pitch:
115 angle = msg.value;
116 if (!toAngleDegree(angle, msg.unit, angle))
117 return false; // Invalid unit
118 setPitch(angle);
119 return true;
120 case MessageContent::Roll:
121 angle = msg.value;
122 if (!toAngleDegree(angle, msg.unit, angle))
123 return false; // Invalid unit
124 setRoll(angle);
125 return true;
126 case MessageContent::Yaw:
127 angle = msg.value;
128 if (!toAngleDegree(angle, msg.unit, angle))
129 return false; // Invalid unit
130 setYaw(angle);
131 return true;
132 default:
133 return false; // Unhandled message content
134 }
135 }
136
137 std::vector<MessageContent> getControls() const override {
138 return {MessageContent::Throttle, MessageContent::Pitch, MessageContent::Roll,
139 MessageContent::Yaw};
140 }

142 MotorT& getMotor(QuadrotorMotorNo motor) { return motors_[motor]; }
143
144 protected:
145 MotorT motors_[4];
146 float motorGain_[4] = {1.0f, 1.0f, 1.0f, 1.0f};
147 float throttle_ = 0;
148 float roll_ = 0;
149 float pitch_ = 0;
150 float yaw_ = 0;
151
152 /**
153 * @brief Update all motors based on throttle, roll, pitch, and yaw.
154 *
155 * This is a simple mixer for an X-configuration quadrotor:
156 * - m1: front left
157 * - m2: front right
158 * - m3: rear left
159 * - m4: rear right
160 *
161 * Each control input is in percent (-100 to 100).
162 */
163 void updateMotors() {
164 // Basic mixing: throttle + pitch/roll/yaw
165 int m[4];
166 m[0] = throttle_ + pitch_ + roll_ - yaw_;
167 m[1] = throttle_ + pitch_ - roll_ + yaw_;
168 m[2] = throttle_ - pitch_ + roll_ + yaw_;
169 m[3] = throttle_ - pitch_ - roll_ - yaw_;
170 for (int i = 0; i < 4; ++i) {
171 int calibrated = static_cast<int>(m[i] * motorGain_[i]);
172 motors_[i].setSpeed(constrain(calibrated, 0, 100));
173 }
174 // publish motor speeds as messages for telemetry
175 for (int i = 0; i < 4; ++i) {
176 Message<float> msg(MessageContent::MotorSpeed, m[i], Unit::Percent);
177 msg.origin = MessageOrigin::Motor;
178 msg.origin_id = i; // Motor index
179 sendMessage(msg);
180 }
181 }
182};
