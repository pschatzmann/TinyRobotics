7// Reads a line from the stream into a std::stream. Returns true if a line was
8// read, false on EOF.
9bool readLine(Stream& s, std::string& out) {
10 out.clear();
11 while (true) {
12 int c = s.read();
13 if (c < 0) {
14 // If nothing read, return false to signal EOF
15 return !out.empty();
16 }
17 if (c == '\n') break;
18 out += static_cast<char>(c);
19 }
20 return true;
21}
