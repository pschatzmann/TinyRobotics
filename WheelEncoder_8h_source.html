12/**
13 * @class WheelEncoder
14 * @ingroup sensors
15 * @brief Measures wheel rotation and computes distance and speed using encoder
16 * ticks for mobile robots.
17 *
18 * The WheelEncoder class provides an interface for tracking the movement of a
19 * robot's wheel using an incremental encoder. It accumulates encoder ticks to
20 * estimate the distance traveled and calculates speed based on tick timing.
21 *
22 * Key features:
23 * - Configurable wheel diameter and ticks per revolution for accurate distance
24 * estimation.
25 * - Periodic reporting of distance and speed via the MessageSource interface.
26 * - Slip calibration support to compensate for wheel slip or surface effects.
27 * - Designed for use with interrupt-driven tick updates (call setTick() in your
28 * ISR).
29 *
30 * Usage:
31 * 1. Create a WheelEncoder instance and configure the wheel diameter and ticks
32 * per revolution.
33 * 2. Call begin() to reset and start periodic reporting.
34 * 3. In your encoder interrupt handler, call setTick() to update the encoder
35 * state.
36 * 4. Use getDistanceM(), getSpeedMPS(), or getDistance() to retrieve odometry
37 * data.
38 * 5. Optionally, calibrate slip using calibrateSlip() if you observe systematic
39 * odometry errors.
40 *
41 * Example:
42 * @code
43 * WheelEncoder encoder;
44 * encoder.setWheelDiameter(0.065); // 65mm wheel
45 * encoder.setTicksPerRevolution(20);
46 * encoder.begin();
47 * // In your interrupt:
48 * encoder.setTick();
49 * // In your main loop:
50 * float distance = encoder.getDistanceM();
51 * float speed = encoder.getSpeedMPS();
52 * @endcode
53 *
54 * This class is intended for embedded robotics applications (Arduino, ESP32,
55 * etc.) and integrates with the TinyRobotics messaging framework.
56 */
57class WheelEncoder : public MessageSource {
58 public:
59 /**
60 * @brief Default constructor. Wheel diameter and ticks per revolution must be
61 * set before use.
62 */
63 WheelEncoder() = default;
64
65 /**
66 * @brief Constructor with wheel diameter and ticks per revolution.
67 * @param wheelDiameterM The diameter of the wheel (Distance object).
68 * @param ticksPerRevolution Number of encoder ticks per full wheel
69 * revolution.
70 */
71 WheelEncoder(Distance wheelDiameterM, int ticksPerRevolution = 1)
72 : WheelEncoder() {
73 setWheelDiameter(wheelDiameterM);
74 setTicksPerRevolution(ticksPerRevolution);
75 }
76
77 /**
78 * @brief Set the wheel diameter using a Distance object.
79 * @param wheelDiameter The wheel diameter.
80 */
81 void setWheelDiameter(Distance wheelDiameter) {
82 wheelDiameterM = wheelDiameter.getValue(DistanceUnit::M);
83 }
84
85 /**
86 * @brief Set the wheel diameter using a float value and unit.
87 * @param diameter The wheel diameter value.
88 * @param unit The unit of the diameter (default: meters).
89 */
90 void setWheelDiameter(float diameter, DistanceUnit unit = DistanceUnit::M) {
91 setWheelDiameter(Distance(diameter, unit));
92 }
93
94 /**
95 * @brief Set the number of encoder ticks per wheel revolution.
96 * @param ticks Number of ticks per revolution.
97 */
98 void setTicksPerRevolution(int ticks) {
99 ticksPerRevolution = ticks;
100 distancePerTickM =
101 static_cast<float>(M_PI) * wheelDiameterM / ticksPerRevolution;
102 }
103
104 /**
105 * @brief Defines the reporint frequency for distance and speed messages.
106 * @param ms Reporting interval in milliseconds.
107 */
108 void setReportingFrequencyMs(uint16_t ms) {
109 reportingScheduler.begin(ms, sendMessageCB, this);
110 }
111
112 /**
113 * @brief Get the total distance traveled since last reset.
114 * @return Distance object representing the traveled distance in meters.
115 */
116 Distance getDistance() const { return Distance(distanceM, DistanceUnit::M); }
117
118 /**
119 * @brief Get the total distance traveled in meters.
120 * @return Distance in meters.
121 */
122 float getDistanceM() const { return distanceM * slipFactor; }
123
124 /**
125 * @brief Get the total distance traveled in the specified unit.
126 * @param unit The distance unit.
127 * @return Distance in the specified unit.
128 */
129 float getDistance(DistanceUnit unit) const {
130 Distance dist(distanceM, DistanceUnit::M);
131 return dist.getValue(unit);
132 }
133
134 /**
135 * @brief Get the distance corresponding to a given number of ticks (in
136 * meters).
137 * @param ticks Number of encoder ticks.
138 * @return Distance in meters.
139 */
140 float getDistanceForTicksM(size_t ticks) const {
141 return ticks * distancePerTickM;
142 }
143
144 /**
145 * @brief Get the distance for a given number of ticks in the specified unit.
146 * @param ticks Number of encoder ticks.
147 * @param unit The distance unit.
148 * @return Distance in the specified unit.
149 */
150 float getDistanceForTicks(size_t ticks, DistanceUnit unit) const {
151 Distance dist(getDistanceForTicksM(ticks), DistanceUnit::M);
152 return dist.getValue(unit);
153 }
154
155 /**
156 * @brief Calculate speed in meters per second based on distance traveled
157 * since last calculation and elapsed time.
158 * @return Speed in meters per second.
159 */
160 float getSpeedMPS() {
161 float speedMPS = 0;
162 uint32_t currentTimeMs = millis();
163 uint32_t elapsedTimeMs = currentTimeMs - lastSpeedCalcTimeMs;
164 if (elapsedTimeMs > 0) {
165 float distanceTraveledM = (distanceM - lastDistanceM) * slipFactor;
166 speedMPS = distanceTraveledM / (elapsedTimeMs / 1000.0f);
167 lastSpeedCalcTimeMs = currentTimeMs;
168 lastDistanceM = distanceM;
169 }
170 return speedMPS;
171 }
172
173 /**
174 * @brief Resets the encoder counts and distance.
175 *
176 * Also starts periodic reporting if not already active.
177 * @return true if the encoder is properly configured (wheel diameter and
178 * ticks per revolution are set).
179 */
180 bool begin() {
181 distanceM = 0;
182 lastSpeedCalcTimeMs = millis();
183 // Default to 1 second reporting if not already set
184 if (!reportingScheduler.isActive()) {
185 setReportingFrequencyMs(1000);
186 }
187 return wheelDiameterM > 0 && ticksPerRevolution > 0;
188 }
189
190 /**
191 * @brief To be called by the pin interrupt handler when a tick is detected.
192 *
193 * Updates the distance and triggers reporting if needed.
194 */
195 void setTick() {
196 distanceM += distancePerTickM;
197 reportingScheduler.run();
198 }
199
200 /**
201 * @brief Trigger sending messages for distance and speed.
202 *
203 * Sends the current distance and speed as messages.
204 */
205 void sendMessage() {
206 Message<float> msg(MessageContent::Distance, getDistanceM(), Unit::Meters);
207 msg.source = MessageOrigin::Sensor;
208 MessageSource::sendMessage(msg);
209
210 Message<float> speedMsg(MessageContent::Speed, getSpeedMPS(),
211 Unit::MetersPerSecond);
212 speedMsg.source = MessageOrigin::Sensor;
213 MessageSource::sendMessage(speedMsg);
214 }
215
216 void end() { reportingScheduler.end(); }
217
218 /**
219 * @brief Set the slip correction factor.
220 *
221 * The slip factor is used to correct encoder-based distance and speed
222 * estimates. It should be set to (actual_distance / encoder_distance) after
223 * calibration. Default is 1.0 (no slip).
224 * @param slipFactor The slip correction factor (typically <= 1.0).
225 */
226 void setSlipFactor(float slipFactor) { this->slipFactor = slipFactor; }
227
228 /**
229 * @brief Calibrate slip by providing a known actual distance.
230 *
231 * Call this after moving the robot a known distance and accumulating encoder
232 * ticks. The slip factor will be set to (actual_distance / encoder_distance).
233 * @param actualDistanceM The actual distance traveled in meters.
234 */
235 void calibrateSlip(float actualDistanceM) {
236 float encoderDistance = getRawDistanceM();
237 if (encoderDistance > 0) {
238 slipFactor = actualDistanceM / encoderDistance;
239 }
240 }
241
242 /**
243 * @brief Get the slip correction factor.
244 * @return The slip factor (1.0 = no slip, <1.0 = slip present).
245 */
246 float getSlipFactor() const { return slipFactor; }
247
248 /**
249 * @brief Get the raw (uncorrected) distance in meters from the encoder.
250 * @return Raw encoder distance in meters.
251 */
252 float getRawDistanceM() const { return distanceM; }
253
254 protected:
255 volatile float distanceM = 0.0f;
256 float wheelDiameterM = 0.0f;
257 float distancePerTickM = 0.0f;
258 float lastDistanceM = 0.0f;
259 float slipFactor = 1.0f;
260 uint32_t ticksPerRevolution = 0;
261 uint32_t lastSpeedCalcTimeMs = 0;
262 Scheduler reportingScheduler;
263
264 /**
265 * @brief Static callback for the reporting scheduler to send messages.
266 * @param ref Pointer to the WheelEncoder instance.
267 */
268 static void sendMessageCB(void* ref) {
269 WheelEncoder* encoder = static_cast<WheelEncoder*>(ref);
270 encoder->sendMessage();
271 }
272};
