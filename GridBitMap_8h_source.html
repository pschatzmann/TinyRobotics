11/**
12 * @class GridMapBitVector
13 * @ingroup maps
14 * @brief A grid map using two bit vectors to represent CellState efficiently.
15 *
16 * The GridMapBitVector class models the environment as a regular grid of cells,
17 * each storing a state (FREE, OCCUPIED, UNKNOWN) using two std::vector<bool>
18 * for memory efficiency. So each cell occupies only 2 buts!
19 *
20 * CellState encoding:
21 * - 00: FREE
22 * - 01: OCCUPIED
23 * - 10: UNKNOWN
24 *
25 * @tparam T Numeric type for coordinates (default: float)
26 */

61 bool worldToCell(float wx, float wy, Cell& cell) const {
62 cell.cx = static_cast<int>((wx - origin.x) / resolution);
63 cell.cy = static_cast<int>((wy - origin.y) / resolution);

72 }

74 // Get cell state by index
75 bool getCell(int cx, int cy, CellState& result) const {
76 if (cx < 0 || cx >= xCount || cy < 0 || cy >= yCount) return false;
77 size_t idx = cy * xCount + cx;
78 if (occupied[idx]) {
79 result = CellState::OCCUPIED;
80 } else if (free[idx]) {
81 result = CellState::FREE;
82 } else {
83 result = CellState::UNKNOWN;
84 }
85 return true;
86 }
87
88 // Set cell state by index
89 void setCell(int cx, int cy, CellState value) {
90 if (cx < 0 || cx >= xCount || cy < 0 || cy >= yCount) return;
91 size_t idx = cy * xCount + cx;
92 switch (value) {
93 case CellState::FREE:
94 occupied[idx] = false;
95 free[idx] = true;
96 break;
97 case CellState::OCCUPIED:
98 occupied[idx] = true;
99 free[idx] = false;
100 break;
101 case CellState::UNKNOWN:
102 occupied[idx] = false;
103 free[idx] = false;
104 break;
105 }
106 }

108 // Get cell state by coordinate
109 bool getCell(const Coordinate<T>& coord, CellState& result) const {
110 Cell cell;
111 if (worldToCell(coord.x, coord.y, cell)) {
112 return getCell(cell.cx, cell.cy, result);

114 return false;
115 }

163 /// Determine all neighboring cells (8-connected) for a given cell coordinate.
164 std::vector<Cell> getNeighborCells(const Coordinate<T> from) const {
165 Cell cell;
166 worldToCell(from.x, from.y, cell);
167 int cx = static_cast<int>(cell.cx);
168 int cy = static_cast<int>(cell.cy);
169 std::vector<Cell> neighbors;
170 if (cx < xCount - 1)
171 neighbors.push_back(
172 {static_cast<size_t>(cx + 1), static_cast<size_t>(cy)});
173 if (cx > 0)
174 neighbors.push_back(
175 {static_cast<size_t>(cx - 1), static_cast<size_t>(cy)});
176 if (cy < yCount - 1)
177 neighbors.push_back(
178 {static_cast<size_t>(cx), static_cast<size_t>(cy + 1)});
179 if (cy > 0)
180 neighbors.push_back(
181 {static_cast<size_t>(cx), static_cast<size_t>(cy - 1)});
182
183 if (cx < xCount - 1 && cy < yCount - 1)
184 neighbors.push_back(
185 {static_cast<size_t>(cx + 1), static_cast<size_t>(cy + 1)});
186 if (cx > 0 && cy < yCount - 1)
187 neighbors.push_back(
188 {static_cast<size_t>(cx - 1), static_cast<size_t>(cy + 1)});
189 if (cx < xCount - 1 && cy > 0)
190 neighbors.push_back(
191 {static_cast<size_t>(cx + 1), static_cast<size_t>(cy - 1)});
192 if (cx > 0 && cy > 0)
193 neighbors.push_back(
194 {static_cast<size_t>(cx - 1), static_cast<size_t>(cy - 1)});
