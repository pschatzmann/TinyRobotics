11 * @class GridMapBitVector
12 * @ingroup maps
13 * @brief A grid map using two bit vectors to represent CellState efficiently.
14 *
15 * The GridMapBitVector class models the environment as a regular grid of cells,
16 * each storing a state (FREE, OCCUPIED, UNKNOWN) using two std::vector<bool>
17 * for memory efficiency. So each cell occupies only 2 buts!
18 *
19 * CellState encoding:
20 * - 00: FREE
21 * - 01: OCCUPIED
22 * - 10: UNKNOWN
23 *
24 * @tparam T Numeric type for coordinates (default: float)
25 */

57 bool worldToCell(float wx, float wy, Cell& cell) const {
58 cell.cx = static_cast<int>((wx - origin.x) / resolution);
59 cell.cy = static_cast<int>((wy - origin.y) / resolution);

70 // Get cell state by index
71 bool getCell(int cx, int cy, CellState& result) const {
72 if (cx < 0 || cx >= xCount || cy < 0 || cy >= yCount) return false;
73 size_t idx = cy * xCount + cx;
74 if (occupied[idx]) {
75 result = CellState::OCCUPIED;
76 } else if (free[idx]) {
77 result = CellState::FREE;
78 } else {
79 result = CellState::UNKNOWN;
80 }
81 return true;

83
84 // Set cell state by index
85 void setCell(int cx, int cy, CellState value) {
86 if (cx < 0 || cx >= xCount || cy < 0 || cy >= yCount) return;
87 size_t idx = cy * xCount + cx;
88 switch (value) {
89 case CellState::FREE:
90 occupied[idx] = false;
91 free[idx] = true;
92 break;
93 case CellState::OCCUPIED:
94 occupied[idx] = true;
95 free[idx] = false;
96 break;
97 case CellState::UNKNOWN:
98 occupied[idx] = false;
99 free[idx] = false;
100 break;
101 }
102 }
103
104 // Get cell state by coordinate
105 bool getCell(const Coordinate<float>& coord, CellState& result) const {
106 Cell cell;
107 if (worldToCell(coord.x, coord.y, cell)) {
108 return getCell(cell.cx, cell.cy, result);
109 }
110 return false;
111 }

113 // Set cell state by coordinate

115 Cell cell;

117 setCell(cell.cx, cell.cy, value);
118 }
119 }
