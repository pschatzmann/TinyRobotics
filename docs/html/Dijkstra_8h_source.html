47 Path<Node> findPath(const MapType& map, const Node& start, const Node& goal) {
48 using Pair = std::pair<float, Node>;
49 std::priority_queue<Pair, std::vector<Pair>, std::greater<Pair>> open;
50 std::unordered_map<Node, float> cost_so_far;
51 std::unordered_map<Node, Node> came_from;
52
53 open.emplace(0.0f, start);
54 cost_so_far[start] = 0.0f;
55 came_from[start] = start;
56
57 while (!open.empty()) {
58 Node current = open.top().second;
59 open.pop();
60
61 if (current == goal) break;
62
63 for (const Node& next : map.getNeighbors(current)) {
64 if (!valid_cb(next, ref)) continue;
65 float new_cost = cost_so_far[current] + cost_cb(current, next, ref);
66 if (!cost_so_far.count(next) || new_cost < cost_so_far[next]) {
67 cost_so_far[next] = new_cost;
68 came_from[next] = current;
69 open.emplace(new_cost, next);
70 }
71 }
72 }
73
74 // Reconstruct path
75 Path<Node> path;
76 if (!came_from.count(goal)) return path;
77 for (Node at = goal; at != start; at = came_from[at]) {
78 path.addWaypoint(at);
79 }
80 path.addWaypoint(start);
81 path.reverse();
82 return path;
83 }
84
85 protected:

87 ValidCallback valid_cb = defaultValid;
88 void* ref = this;
