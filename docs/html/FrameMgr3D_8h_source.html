19struct Transform3D {
20 float tx = 0, ty = 0, tz = 0;
21 float qx = 0, qy = 0, qz = 0, qw = 1;
22
23 static Transform3D identity() { return {}; }
24
25 // Quaternion multiplication and translation composition
26 Transform3D operator*(const Transform3D& other) const {
27 Transform3D r;
28 // Quaternion multiplication
29 r.qw = qw * other.qw - qx * other.qx - qy * other.qy - qz * other.qz;
30 r.qx = qw * other.qx + qx * other.qw + qy * other.qz - qz * other.qy;
31 r.qy = qw * other.qy - qx * other.qz + qy * other.qw + qz * other.qx;
32 r.qz = qw * other.qz + qx * other.qy - qy * other.qx + qz * other.qw;
33
34 // Rotate other's translation by this quaternion and add to this translation
35 float x = other.tx, y = other.ty, z = other.tz;
36 float ix = qw * x + qy * z - qz * y;
37 float iy = qw * y + qz * x - qx * z;
38 float iz = qw * z + qx * y - qy * x;
39 float iw = -qx * x - qy * y - qz * z;
40 r.tx = tx + (ix * qw + iw * -qx + iy * -qz - iz * -qy);
41 r.ty = ty + (iy * qw + iw * -qy + iz * -qx - ix * -qz);
42 r.tz = tz + (iz * qw + iw * -qz + ix * -qy - iy * -qx);
43
44 return r;
45 }
46
47 // Inverse of the transform
48 Transform3D inverse() const {
49 Transform3D r;
50 // Inverse rotation (conjugate)
51 r.qx = -qx;
52 r.qy = -qy;
53 r.qz = -qz;
54 r.qw = qw;
55 // Inverse translation
56 float x = -tx, y = -ty, z = -tz;
57 float ix = r.qw * x + r.qy * z - r.qz * y;
58 float iy = r.qw * y + r.qz * x - r.qx * z;
59 float iz = r.qw * z + r.qx * y - r.qy * x;
60 float iw = -r.qx * x - r.qy * y - r.qz * z;
61 r.tx = (ix * r.qw + iw * -r.qx + iy * -r.qz - iz * -r.qy);
62 r.ty = (iy * r.qw + iw * -r.qy + iz * -r.qx - ix * -r.qz);
63 r.tz = (iz * r.qw + iw * -r.qz + ix * -qy - iy * -qx);
64 return r;
65 }

73 std::array<float, 3> apply(const std::array<float, 3>& point) const {
74 // Quaternion rotation
75 float x = point[0], y = point[1], z = point[2];
76 // Quaternion-vector multiplication (q * v * q^-1)
77 float qx2 = qx + qx, qy2 = qy + qy, qz2 = qz + qz;
78 float xx = qx * qx2, yy = qy * qy2, zz = qz * qz2;
79 float xy = qx * qy2, xz = qx * qz2, yz = qy * qz2;
80 float wx = qw * qx2, wy = qw * qy2, wz = qw * qz2;
81 float rx = (1 - (yy + zz)) * x + (xy - wz) * y + (xz + wy) * z;
82 float ry = (xy + wz) * x + (1 - (xx + zz)) * y + (yz - wx) * z;
83 float rz = (xz - wy) * x + (yz + wx) * y + (1 - (xx + yy)) * z;
84 return {rx + tx, ry + ty, rz + tz};
85 }

144 // Returns the transform from 'from' to 'to'
145 Transform3D getTransform(const Frame3D& from, const Frame3D& to) const {
146 std::vector<const Frame3D*> fromPath = getPathToRoot(&from);
147 std::vector<const Frame3D*> toPath = getPathToRoot(&to);
148
149 // Find common ancestor
150 int i = fromPath.size() - 1;
151 int j = toPath.size() - 1;
152 while (i >= 0 && j >= 0 && fromPath[i] == toPath[j]) {
153 i--;
154 j--;
155 }
156 i++;
157 j++;
158
159 // Compose transforms up from 'from' to ancestor
160 Transform3D tf_from = Transform3D::identity();
161 for (int k = 0; k < i; ++k) {
162 tf_from = fromPath[k]->getTransform() * tf_from;
163 }
164 // Compose transforms up from 'to' to ancestor
165 Transform3D tf_to = Transform3D::identity();
166 for (int k = 0; k < j; ++k) {
167 tf_to = toPath[k]->getTransform() * tf_to;
168 }
169 // The transform from 'from' to 'to' is tf_to.inverse() * tf_from
170 return tf_to.inverse() * tf_from;
171 }

199 GPSCoordinate toGPS(const Frame3D& frame,
200 const std::array<float, 3>& local) const {
201 if (!p_gpsParent) return gpsCoordinate;
202 // Get transform from local frame to GPS parent frame
203 Transform3D tf = getTransform(frame, *p_gpsParent);
204 // Apply the transform to the local coordinate to get position in GPS parent
205 // frame
206 std::array<float, 3> xyz = tf.apply(local);
207 // Rotate by gpsRotationDeg to align with GPS north (in x/y plane)
208 float theta_rad = gpsRotationDeg * M_PI / 180.0f;
209 float cos_theta = cos(theta_rad);
210 float sin_theta = sin(theta_rad);
211 float dx = cos_theta * xyz[0] - sin_theta * xyz[1];
212 float dy = sin_theta * xyz[0] + cos_theta * xyz[1];
213 float dz = xyz[2];
214 // Use GPSCoordinate::navigate for conversion (distance in meters, bearing
215 // in degrees)
216 float distance = sqrt(dx * dx + dy * dy);
217 float bearing_deg = atan2(dx, dy) * 180.0 / M_PI;
218 return gpsCoordinate.navigate(distance, bearing_deg, dz);
219 }

221 protected:
222 // Returns the path from the given frame up to the root (including self)
223 std::vector<const Frame3D*> getPathToRoot(const Frame3D* frame) const {
224 std::vector<const Frame3D*> path;
225 while (frame) {
226 path.push_back(frame);
227 frame = frame->getParent();

229 return path;
230 }
