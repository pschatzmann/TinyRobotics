145 void updateSpeed(float distanceFromStartM, float distanceToTarget,
146 float currentSpeedKmh) {
147 // Compute a desired speed profile: slow down as you approach the target
148 float minSpeed = 0.0f;
149 float maxSpeed = maxSpeedKmh;
150 desiredSpeedKmh = maxSpeed; // * (distanceToTarget / accelDistanceM);
151 if (distanceToTarget < accelDistanceM) {
152 desiredSpeedKmh = maxSpeed * (distanceToTarget / accelDistanceM);
153 }
154 if (distanceFromStartM < accelDistanceM) {
155 desiredSpeedKmh = maxSpeed * (distanceFromStartM / accelDistanceM);
156 }
157 if (desiredSpeedKmh > maxSpeed) desiredSpeedKmh = maxSpeed;
158 if (desiredSpeedKmh < minSpeed) desiredSpeedKmh = minSpeed;
159 }

184 float computeHeadingError(float desiredHeading, float currentHeading) {
185 float headingError = desiredHeading - currentHeading;
186 if (headingError > 180) headingError -= 360;
187 if (headingError < -180) headingError += 360;

194 Message<float> msg;

196 switch (control) {
197 case MessageContent::Throttle:
198 msg.content = MessageContent::Throttle;
199 // Use PID to map speed to throttle
200 msg.value = pidSpeed_.calculate(desiredSpeedKmh, currentSpeedKmh);
201 break;
202 case MessageContent::SteeringAngle:
203 msg.content = MessageContent::SteeringAngle;
204 msg.value = pidSteering_.calculate(0.0f, headingError);
205 break;
206 default:

208 }
