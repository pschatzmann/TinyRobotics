38 virtual int readArray(T data[], int len) {
39 if (data == nullptr) {
40 LOGE("NPE");
41 return 0;
42 }
43 int lenResult = min(len, available());
44 for (int j = 0; j < lenResult; j++) {
45 read(data[j]);
46 }
47 LOGD("readArray %d -> %d", len, lenResult);
48 return lenResult;
49 }
50
52 virtual int clearArray(int len) {
53 int lenResult = min(len, available());
54 T dummy[lenResult];
55 readArray(dummy, lenResult);
56 return lenResult;
57 }
58
60 virtual int writeArray(const T data[], int len) {
61 // LOGD("%s: %d", LOG_METHOD, len);
62 // CHECK_MEMORY();
63
64 int result = 0;
65 for (int j = 0; j < len; j++) {
66 if (!write(data[j])) {
67 break;
68 }
69 result = j + 1;
70 }
71 // CHECK_MEMORY();
72 LOGD("writeArray %d -> %d", len, result);
73 return result;
74 }

77 virtual int writeArrayOverwrite(const T data[], int len) {
78 int to_delete = len - availableForWrite();
79 if (to_delete > 0) {
80 clearArray(to_delete);

82 return writeArray(data, len);
83 }

207 int len_available = available();
208 if (len > len_available) {
209 len = len_available;
210 }
211 memcpy(data, buffer.data() + current_read_pos, len);
212 return len;
213 }

216 int clearArray(int len) override {
217 int len_available = available();
218 if (len > available()) {
219 reset();
220 return len_available;
221 }
222 current_read_pos += len;
223 len_available -= len;
224 memmove(buffer.data(), buffer.data() + current_read_pos,
225 len_available * sizeof(T));
226 current_read_pos = 0;
227 current_write_pos = len_available;

229 if (is_clear_with_zero) {

231 buffer.size() - current_write_pos);

312 bool read(T& result) override {
313 if (isEmpty()) {
314 return false;
315 }
316
317 result = _aucBuffer[_iTail];
318 _iTail = nextIndex(_iTail);
319 _numElems--;
320
321 return true;
322 }
323
324 // peeks the actual entry from the buffer
325 bool peek(T& result) override {
326 if (isEmpty()) {
327 return false;
328 }
329
330 result = _aucBuffer[_iTail];
331 return true;
332 }
333
334 virtual int peekArray(T* data, int n) {
335 if (isEmpty()) return -1;
336 int result = 0;
337 int count = _numElems;
338 int tail = _iTail;
339 for (int j = 0; j < n; j++) {
340 data[j] = _aucBuffer[tail];
341 tail = nextIndex(tail);
342 count--;
343 result++;
344 if (count == 0) break;
345 }
346 return result;
347 }
348
349 // checks if the buffer is full
350 virtual bool isFull() override { return available() == max_size; }
351
352 bool isEmpty() { return available() == 0; }
353
354 // write add an entry to the buffer
355 virtual bool write(T data) override {
356 bool result = false;
357 if (!isFull()) {
358 _aucBuffer[_iHead] = data;
359 _iHead = nextIndex(_iHead);
360 _numElems++;
361 result = true;
362 }
363 return result;
364 }
365
366 // clears the buffer
367 virtual void reset() override {
368 _iHead = 0;
369 _iTail = 0;
370 _numElems = 0;
371 }
372
373 // provides the number of entries that are available to read
374 virtual int available() override { return _numElems; }
375
376 // provides the number of entries that are available to write
377 virtual int availableForWrite() override { return (max_size - _numElems); }
378
379 // returns the address of the start of the physical read buffer
380 virtual T* address() override { return _aucBuffer.data(); }
381
382 virtual bool resize(int len) {
383 if (max_size != len && len > 0) {
384 LOGI("resize: %d", len);
385 _aucBuffer.resize(len);
386 max_size = len;
387 }

389 }

402 if (max_size == 0) return 0;
