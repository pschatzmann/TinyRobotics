59 float getSpeedMPS(float throttlePercent) const {
60 if (calibrationData.empty()) return 0.0f;
61 // Clamp below
62 if (throttlePercent <= calibrationData.front().first)
63 return calibrationData.front().second;
64 // Clamp above
65 if (throttlePercent >= calibrationData.back().first)
66 return calibrationData.back().second;
67 // Find interval
68 for (size_t i = 1; i < calibrationData.size(); ++i) {
69 float t0 = calibrationData[i - 1].first;
70 float t1 = calibrationData[i].first;
71 if (throttlePercent >= t0 && throttlePercent <= t1) {
72 float s0 = calibrationData[i - 1].second;
73 float s1 = calibrationData[i].second;
74 float alpha = (throttlePercent - t0) / (t1 - t0);
75 return s0 + alpha * (s1 - s0);
76 }
77 }
78 // Should not reach here
79 return 0.0f;
80 }
