59/// Convert a value from the given unit to degrees.
60inline bool toAngleDegree(DEFAULT_TYPE in, Unit unit, AngleDeg& out) {
61 switch (unit) {
62 case Unit::AngleDegree:
63 out = in;
64 return true;
65 case Unit::AngleRadian:
66 out = in * 180.0 / M_PI;
67 return true;
68 default:
69 return false; // Unsupported unit for angle
70 }
