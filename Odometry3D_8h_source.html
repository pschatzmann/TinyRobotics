12/**
13 * @class Odometry3D
14 * @ingroup odometry
15 * @brief Tracks 3D position and orientation of a robot using velocity and
16 * angular rates.
17 *
18 * This class provides simple 3D odometry for mobile robots, such as drones or
19 * underwater vehicles. It integrates linear velocity and angular velocity over
20 * time to estimate the robot's position (x, y, z) and orientation (yaw, pitch,
21 * roll) in meters and radians.
22 *
23 * ## Inputs
24 * - Linear velocity (vx, vy, vz) in m/s (robot frame)
25 * - Angular velocity (wx, wy, wz) in rad/s (robot frame)
26 * - Time delta (in milliseconds, or uses millis() if not provided)
27 *
28 * ## Outputs
29 * - 3D position (x, y, z) in meters
30 * - 3D orientation (yaw, pitch, roll) in radians
31 *
32 * ## Limitations
33 * - Assumes no slip or drift
34 * - Subject to integration drift over time
35 *
36 * @author TinyRobotics contributors
37 * @date 2026-03-31
38 */
39class Odometry3D {
40 public:
41 Odometry3D() = default;
42
43 /**
44 * @brief Initialize the odometry state.
45 * @param initialPosition The starting position (x, y, z) in meters.
46 * @param initialOrientation The starting orientation (yaw, pitch, roll) in
47 * radians.
48 * @return true on success
49 */
50 bool begin(Coordinate<float> initialPosition = {0, 0, 0},
51 Orientation3D initialOrientation = Orientation3D()) {
52 position = initialPosition;
53 orientation = initialOrientation;
54 totalDistance = 0.0f;
55 lastUpdateTimeMs = 0;
56 return true;
57 }
58
59 /**
60 * @brief Update the odometry state with new velocities and angular rates.
61 * @param vx Linear velocity in x (m/s, robot frame)
62 * @param vy Linear velocity in y (m/s, robot frame)
63 * @param vz Linear velocity in z (m/s, robot frame)
64 * @param wx Angular velocity in x (rad/s, robot frame)
65 * @param wy Angular velocity in y (rad/s, robot frame)
66 * @param wz Angular velocity in z (rad/s, robot frame)
67 * @param deltaTimeMs Time since last update in milliseconds
68 */
69 void update(float vx, float vy, float vz, float wx, float wy, float wz,
70 uint32_t deltaTimeMs) {
71 float dt = static_cast<float>(deltaTimeMs) / 1000.0f;
72 // Integrate orientation (Euler angles, simple integration)
73 orientation.roll += wx * dt;
74 orientation.pitch += wy * dt;
75 orientation.yaw += wz * dt;
76 orientation.wrap();
77 // Rotate velocity to world frame (using current orientation)
78 float cr = cos(orientation.roll), sr = sin(orientation.roll);
79 float cp = cos(orientation.pitch), sp = sin(orientation.pitch);
80 float cy = cos(orientation.yaw), sy = sin(orientation.yaw);
81 // Rotation matrix (ZYX convention)
82 float vx_world = cy * cp * vx + (cy * sp * sr - sy * cr) * vy +
83 (cy * sp * cr + sy * sr) * vz;
84 float vy_world = sy * cp * vx + (sy * sp * sr + cy * cr) * vy +
85 (sy * sp * cr - cy * sr) * vz;
86 float vz_world = -sp * vx + cp * sr * vy + cp * cr * vz;
87 // Integrate velocity for position
88 float dx = vx_world * dt;
89 float dy = vy_world * dt;
90 float dz = vz_world * dt;
91 position.x += dx;
92 position.y += dy;
93 position.z += dz;
94 totalDistance += sqrt(dx * dx + dy * dy + dz * dz);
95 lastDelta = Distance3D(dx, dy, dz, DistanceUnit::M);
96 }

109 * @brief Update with automatic time delta (uses millis()).
110 */
111 void update(float vx, float vy, float vz, float wx, float wy, float wz) {
112 uint32_t now = millis();
113 uint32_t deltaTimeMs = now - lastUpdateTimeMs;
114 if (lastUpdateTimeMs > 0) update(vx, vy, vz, wx, wy, wz, deltaTimeMs);
115 lastUpdateTimeMs = now;
116 }
