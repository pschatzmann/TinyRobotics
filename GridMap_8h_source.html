12/**
13 * @class GridMap
14 * @ingroup maps
15 * @brief A grid map for spatial representation, navigation, and planning.
16 *
17 * The GridMap class models the environment as a regular grid of cells, each
18 * storing a state (e.g., occupancy, height, or other user-defined data). It
19 * supports:
20 * - Conversion between world coordinates and grid cell indices
21 * - Efficient cell state access and modification
22 * - Binary Bayesian updates for probabilistic occupancy mapping
23 * - Custom cell validity logic (e.g., for dynamic obstacles)
24 * - Extraction of valid neighbor cells for pathfinding (8-connected)
25 *
26 * Typical use cases include:
27 * - Robot navigation and path planning
28 * - Occupancy grid mapping (SLAM)
29 * - Environment modeling for simulation or AI
30 *
31 * Features:
32 * - Template parameter for cell state type (e.g., occupancy, height)
33 * - World-to-cell and cell-to-world coordinate conversion
34 * - Fast access to cell state by index or coordinate
35 * - Probabilistic update using log-odds for sensor fusion
36 * - Customizable cell validity via callback (see setValidityCallback)
37 * - Customizable cell state conversion via callback (see
38 * setCellStateCallback)
39 * - Neighbor cell extraction for graph-based search
40 *
41 * Callback Notes:
42 * - Validity Callback: Use setValidityCallback to provide custom logic for
43 * determining if a cell is valid (e.g., for dynamic obstacles or special
44 * terrain). The callback receives cell indices and an optional reference
45 * pointer.
46 * - Cell State Callback: Use setCellStateCallback to provide custom logic for
47 * converting StateType to CellState (e.g., for visualization or compatibility
48 * with algorithms expecting CellState). The callback receives the cell value
49 * and an optional reference pointer.
50 * - Use setReference to set the reference pointer passed to both callbacks.
51 *
52 * Example:
53 * @code
54 * GridMap<CellState> map(100, 100, 0.1f); // 10m x 10m, 10cm resolution
55 * Coordinate<float> pos(1.2, 3.4);
56 * map.setCell(pos, CellState::OCCUPIED);
57 * auto state = map.getCell(5, 5);
58 *
59 * // For GridMap<float>, set a callback to convert float to CellState
60 * map.setCellStateCallback([](const float& v, void*) {
61 * return v > 0.5f ? CellState::OCCUPIED : CellState::FREE;
62 * });
63 * @endcode
64 *
65 * @tparam StateType The type stored in each cell (e.g., occupancy, height)
66 * @tparam T Numeric type for coordinates (default: float)
67 */

89 /// World to cell conversion
90 bool worldToCell(DistanceM wx, DistanceM wy, Cell& cell) const {
91 cell.cx = static_cast<int>(std::floor((wx - origin.x) / resolution));
92 cell.cy = static_cast<int>(std::floor((wy - origin.y) / resolution));
93 return (cell.cx >= 0 && cell.cx < xCount && cell.cy >= 0 &&
94 cell.cy < yCount);

161 /// Determine all neighboring cells (8-connected) for a given cell coordinate.
162 std::vector<Cell> getNeighborCells(const Coordinate<DistanceM> from) const {
163 Cell cell;
164 worldToCell(from.x, from.y, cell);
165 int cx = static_cast<int>(cell.cx);
166 int cy = static_cast<int>(cell.cy);
167 std::vector<Cell> neighbors;
168 if (cx < xCount - 1)
169 neighbors.push_back(
170 {static_cast<size_t>(cx + 1), static_cast<size_t>(cy)});
171 if (cx > 0)
172 neighbors.push_back(
173 {static_cast<size_t>(cx - 1), static_cast<size_t>(cy)});
174 if (cy < yCount - 1)
175 neighbors.push_back(
176 {static_cast<size_t>(cx), static_cast<size_t>(cy + 1)});
177 if (cy > 0)
178 neighbors.push_back(
179 {static_cast<size_t>(cx), static_cast<size_t>(cy - 1)});
180
181 if (cx < xCount - 1 && cy < yCount - 1)
182 neighbors.push_back(
183 {static_cast<size_t>(cx + 1), static_cast<size_t>(cy + 1)});
184 if (cx > 0 && cy < yCount - 1)
185 neighbors.push_back(
186 {static_cast<size_t>(cx - 1), static_cast<size_t>(cy + 1)});
187 if (cx < xCount - 1 && cy > 0)
188 neighbors.push_back(
189 {static_cast<size_t>(cx + 1), static_cast<size_t>(cy - 1)});
190 if (cx > 0 && cy > 0)
191 neighbors.push_back(
192 {static_cast<size_t>(cx - 1), static_cast<size_t>(cy - 1)});
193 return neighbors;
194 }

224 bool isValid(const Coordinate<T>& coord) const {
225 int cx, cy;
226 cx = static_cast<int>((coord.x - origin.x) / resolution);
227 cy = static_cast<int>((coord.y - origin.y) / resolution);
228 bool in_range = cx >= 0 && cx < xCount && cy >= 0 && cy < yCount;
229 if (!in_range) return false; // Out of bounds
230 if (is_valid_cb != nullptr) {
231 return is_valid_cb(cx, cy, reference);
232 }
233 return true; // In bounds, valid by default
234 }

264 static bool isValid(int cx, int cy, void* ref) {

266 StateType result;

268 if constexpr (std::is_same<StateType, CellState>::value) {
