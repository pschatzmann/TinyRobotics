128 std::vector<Cell> getNeighborCells(const Coordinate<DistanceM> from) const {
129 Cell cell;
130 worldToCell(from.x, from.y, cell);
131 int cx = static_cast<int>(cell.cx);
132 int cy = static_cast<int>(cell.cy);
133 std::vector<Cell> neighbors;
134 if (cx < xCount - 1)
135 neighbors.push_back(
136 {static_cast<size_t>(cx + 1), static_cast<size_t>(cy)});
137 if (cx > 0)
138 neighbors.push_back(
139 {static_cast<size_t>(cx - 1), static_cast<size_t>(cy)});
140 if (cy < yCount - 1)
141 neighbors.push_back(
142 {static_cast<size_t>(cx), static_cast<size_t>(cy + 1)});
143 if (cy > 0)
144 neighbors.push_back(
145 {static_cast<size_t>(cx), static_cast<size_t>(cy - 1)});
146
147 if (cx < xCount - 1 && cy < yCount - 1)
148 neighbors.push_back(
149 {static_cast<size_t>(cx + 1), static_cast<size_t>(cy + 1)});
150 if (cx > 0 && cy < yCount - 1)
151 neighbors.push_back(
152 {static_cast<size_t>(cx - 1), static_cast<size_t>(cy + 1)});
153 if (cx < xCount - 1 && cy > 0)
154 neighbors.push_back(
155 {static_cast<size_t>(cx + 1), static_cast<size_t>(cy - 1)});
156 if (cx > 0 && cy > 0)
157 neighbors.push_back(
158 {static_cast<size_t>(cx - 1), static_cast<size_t>(cy - 1)});
159 return neighbors;
160 }
