15/**
16 * @class MotionController2D
17 * @ingroup control
18 * @brief 2D motion controller for path following and vehicle control.
19 *
20 * This class implements a 2D motion controller for robotic vehicles using PID
21 * control for both speed (throttle) and steering. It supports path following,
22 * smooth acceleration/deceleration, and integrates with an IMU for feedback.
23 *
24 * - Uses a configurable PID controller for speed-to-throttle mapping.
25 * - Uses a configurable PID controller for steering angle correction.
26 * - Accepts a path (sequence of waypoints) and drives the vehicle to follow it.
27 * - Automatically slows down as the vehicle approaches the goal.
28 * - Designed for modular use with different vehicle types and IMU sensors.
29 * - Vehicle needs to suscribe to receive control messages (throttle and
30 * steering angle).
31 *
32 * @tparam T Numeric type for calculations (default: float)
33 *
34 * @author Phil Schatzmann
35 */
36template <typename T = float>
37class MotionController2D : public MessageSource {

110 /// Main control loop to be called periodically (e.g., in a timer or main
111 /// loop)
112 void update() {
113 if (!is_active) return;
114 if (!initializeDtFromUpdates()) {
115 return;
116 }
117 Coordinate<DistanceM> currentPos = motionStateSource.getPosition();
118 Coordinate<DistanceM> targetPos = path[0];
119 float currentHeading =
120 motionStateSource.getHeading().getValue(AngleUnit::DEG);
121 float desiredHeading = 0;
122 if (!handleWaypoint(currentPos, targetPos, distanceToTargetM,
123 desiredHeading)) {
124 // new waypoint
125 return;
126 }
127 float headingError = computeHeadingError(desiredHeading, currentHeading);
128 float currentSpeedKmh =
129 motionStateSource.getSpeed().getValue(SpeedUnit::KPH);
130 float distanceFromStartM =
131 startCoordinate.distance(currentPos, DistanceUnit::M);
132 updateSpeed(distanceFromStartM, distanceToTargetM, currentSpeedKmh);
133
134 has_distance = true;
135 sendControlMessages(distanceToTargetM, headingError, currentSpeedKmh);
136 }

183 /// Calculate desired speed based on distance to target and distance from
184 /// start
185 void updateSpeed(float distanceFromStartM, float distanceToTarget,
186 float currentSpeedKmh) {
187 // Compute a desired speed profile: slow down as you approach the target
188 float minSpeedKmh = 0.0f;
189 desiredSpeedKmh = maxSpeedKmh; // * (distanceToTarget / accelDistanceM);
190 if (distanceToTarget < accelDistanceM) {
191 if (distanceToTarget < targetAccuracyM) {
192 desiredSpeedKmh = 0.0f; // Stop at the target
193 } else {
194 desiredSpeedKmh = maxSpeedKmh * (distanceToTarget / accelDistanceM);
195 }
196 }
197 if (distanceFromStartM < accelDistanceM) {
198 // avoid 0 start speed!
199 if (distanceFromStartM == 0)
200 distanceFromStartM = 0.1f; // avoid division by zero
201 desiredSpeedKmh = maxSpeedKmh * (distanceFromStartM / accelDistanceM);
202 }
203 if (desiredSpeedKmh > maxSpeedKmh) desiredSpeedKmh = maxSpeedKmh;
204 if (desiredSpeedKmh < minSpeedKmh) desiredSpeedKmh = minSpeedKmh;
205 }

207 /// Handle the current waypoint: calculate distance and desired heading
208 bool handleWaypoint(const Coordinate<DistanceM>& currentPos,
209 Coordinate<DistanceM>& targetPos,
210 float& distanceToTargetM, float& desiredHeadingDeg) {
211 // Use class variable targetAccuracyM
212 distanceToTargetM = 0.0f;
213 while (distanceToTargetM < targetAccuracyM) {
214 desiredHeadingDeg = currentPos.bearing(targetPos, AngleUnit::DEG);
215 distanceToTargetM = currentPos.distance(targetPos, DistanceUnit::M);
216 if (distanceToTargetM < targetAccuracyM) {
217 path.removeHead();
218 if (path.isEmpty()) {
219 if (onGoalCallback) {
220 onGoalCallback(onGoaldRef);
221 }
222 return false;
223 }
224 targetPos = path[0];
225 return false;
226 }
227 }
228 return true;
229 }

231 float computeHeadingError(float desiredHeading, float currentHeading) {
232 float headingError = desiredHeading - currentHeading;
233 while (headingError > 180) headingError -= 360;
234 while (headingError < -180) headingError += 360;
235 return headingError;
236 }

238 void sendControlMessages(float distanceToTarget, float headingError,
239 float currentSpeedKmh) {
240 Message<float> msg;
241 float throttlePercent =
242 pidSpeed_.calculate(desiredSpeedKmh, currentSpeedKmh);
243 // Clamp throttle to [0, 100]
244 if (throttlePercent > 100.0f) throttlePercent = 100.0f;
245 if (throttlePercent < 0.0f && throttlePercent > -1.0f)
246 throttlePercent = 0.0f;
247 if (throttlePercent > 0.0f && throttlePercent < 1.0f)
248 throttlePercent = 1.0f; // avoid too low throttle
249 if (throttlePercent < 0.0f) throttlePercent = 0.0f;
250 msg.source = MessageOrigin::Autonomy;
251 msg.content = MessageContent::Throttle;
252 msg.unit = Unit::Percent;
253 // Use PID to map speed to throttle
254 msg.value = resultThrottlePercent = throttlePercent;
255 sendMessage(msg);
256 msg.content = MessageContent::SteeringAngle;
257 msg.unit = Unit::AngleDegree;
258 msg.value = resultStreeringAngleDeg =
259 pidSteering_.calculate(0.0f, -headingError);
260 sendMessage(msg);
261 TRLogger.info(
262 "MotionController2D: distanceToTarget=%.2f m, desiredSpeed=%.2f km/h, "
263 "currentSpeed=%.2f km/h, throttle=%.1f%%, headingError=%.1f deg, "
264 "steeringAngle=%.1f deg",
265 distanceToTarget, desiredSpeedKmh, currentSpeedKmh,
266 resultThrottlePercent, headingError, resultStreeringAngleDeg);
267 }

269 /// Handles dt initialization from first 10 updates, but does not block
270 /// control logic
271 bool initializeDtFromUpdates() {
272 unsigned long nowMs = millis();
273 if (dtSetFromUpdates) return true;
274 if (updateCount == 0) {
275 updateStartTimeMs = nowMs;
276 }
277 updateCount++;
278 if (updateCount == 11) {
279 updateEndTimeMs = nowMs;
280 float avgMs = (updateEndTimeMs - updateStartTimeMs) / 10.0f;
281 float dt = avgMs / 1000.0f; // convert ms to seconds
282 // Reconfigure both PIDs with new dt, keep other params
283 pidSpeed_.setDt(dt);
284 pidSteering_.setDt(dt);
285 dtSetFromUpdates = true;
286 }
287 return dtSetFromUpdates;
288 }
