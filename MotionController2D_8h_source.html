21/**
22 * @class MotionController2D
23 * @ingroup control
24 * @brief 2D motion controller for path following and vehicle control.
25 *
26 * This class implements a flexible 2D motion controller for robotic vehicles,
27 * supporting both model-based (feedforward) and feedback (PID) control for
28 * speed (throttle) and steering. It is designed for path following, smooth
29 * acceleration/deceleration, and robust integration with IMU and odometry
30 * sensors.
31 *
32 * **Features:**
33 * - Configurable PID controllers for both speed (throttle) and steering angle
34 * correction.
35 * - Multiple throttle control strategies: Feedforward (model-based), PID
36 * (feedback), or Combined (feedforward + feedback).
37 * - Path following with a sequence of waypoints, including automatic waypoint
38 * advancement and goal detection.
39 * - Smooth acceleration and deceleration profiles based on distance to target
40 * and from start.
41 * - Dynamic time step (dt) initialization for PID controllers based on update
42 * frequency.
43 * - Modular design for use with different vehicle types, IMU sensors, and
44 * coordinate systems.
45 * - Integration with the TinyRobotics message system for sending throttle and
46 * steering commands.
47 * - Customizable goal callback for user-defined actions upon reaching the final
48 * waypoint.
49 *
50 * **Usage Example:**
51 * @code
52 * MotionController2D<> controller(motionState, 10.0f, 2.0f);
53 * controller.setPath(path);
54 * controller.setThrottleMode(ThrottleMode::Combined);
55 * controller.begin();
56 * // In main loop:
57 * controller.update();
58 * @endcode
59 *
60 * **Integration:**
61 * - Use as a standalone controller for path following and vehicle control.
62 * - Integrate with a message bus to receive and send control messages.
63 * - Extend or customize by overriding protected methods or providing custom
64 * callbacks.
65 *
66 * @tparam T Numeric type for calculations (default: float)
67 *
68 * @author Phil Schatzmann
69 */

158 /**
159 * @brief Main control loop to be called periodically (e.g., in a timer or
160 * main loop)
161 */
162 void update() {
163 if (!is_active) return;
164 if (!initializeDtFromUpdates()) {
165 return;
166 }
167 Coordinate<DistanceM> currentPos = motionStateSource.getPosition();
168 Coordinate<DistanceM> targetPos = path[0];
169 float currentHeading =
170 motionStateSource.getHeading().getValue(AngleUnit::DEG);
171 float desiredHeading = 0;
172 if (!handleWaypoint(currentPos, targetPos, distanceToTargetM,
173 desiredHeading)) {
174 // new waypoint
175 return;
176 }
177 float headingError = computeHeadingError(desiredHeading, currentHeading);
178 float currentSpeedKmh =
179 motionStateSource.getSpeed().getValue(SpeedUnit::KPH);
180 float distanceFromStartM =
181 startCoordinate.distance(currentPos, DistanceUnit::M);
182 updateSpeed(distanceFromStartM, distanceToTargetM, currentSpeedKmh);
183
184 has_distance = true;
185 sendControlMessages(distanceToTargetM, headingError, currentSpeedKmh,
186 desiredHeading, currentHeading);
187 }

267 /// Calculate desired speed based on distance to target and distance from
268 /// start
269 void updateSpeed(float distanceFromStartM, float distanceToTarget,
270 float currentSpeedKmh) {
271 // Compute a desired speed profile: slow down as you approach the target
272 float minSpeedKmh = 0.0f;
273 desiredSpeedKmh = maxSpeedKmh; // * (distanceToTarget / accelDistanceM);
274 if (distanceToTarget < accelDistanceM) {
275 if (distanceToTarget < targetAccuracyM) {
276 desiredSpeedKmh = 0.0f; // Stop at the target
277 } else {
278 desiredSpeedKmh = maxSpeedKmh * (distanceToTarget / accelDistanceM);
279 }
280 }
281 if (distanceFromStartM < accelDistanceM) {
282 // avoid 0 start speed!
283 if (distanceFromStartM == 0)
284 distanceFromStartM = 0.1f; // avoid division by zero
285 desiredSpeedKmh = maxSpeedKmh * (distanceFromStartM / accelDistanceM);
286 }
287 if (desiredSpeedKmh > maxSpeedKmh) desiredSpeedKmh = maxSpeedKmh;
288 if (desiredSpeedKmh < minSpeedKmh) desiredSpeedKmh = minSpeedKmh;
289 }

291 /// Handle the current waypoint: calculate distance and desired heading
292 bool handleWaypoint(const Coordinate<DistanceM>& currentPos,
293 Coordinate<DistanceM>& targetPos,
294 float& distanceToTargetM, float& desiredHeadingDeg) {
295 // Use class variable targetAccuracyM
296 distanceToTargetM = 0.0f;
297 while (distanceToTargetM < targetAccuracyM) {
298 desiredHeadingDeg = currentPos.bearing(targetPos, AngleUnit::DEG);
299 distanceToTargetM = currentPos.distance(targetPos, DistanceUnit::M);
300 if (distanceToTargetM < targetAccuracyM) {
301 TRLogger.info(
302 "Reached waypoint at (%.2f m, %.2f m), distance to target: %.2f m",
303 targetPos.x, targetPos.y, distanceToTargetM);
304 path.removeHead();
305 if (path.isEmpty()) {
306 TRLogger.info("Reached final goal!");
307 if (onGoalCallback) {
308 onGoalCallback(onGoalRef);
309 }
310 return false;
311 }
312 targetPos = path[0];
313 return false;
314 }
315 }
316 return true;
317 }

319 float computeHeadingError(float desiredHeading, float currentHeading) {
320 float headingError = desiredHeading - currentHeading;
321 while (headingError > 180) headingError -= 360;
322 while (headingError < -180) headingError += 360;
323 return headingError;
324 }

326 /**
327 * @brief Model-based feedforward throttle estimate: 100% throttle =
328 * maxSpeedKmh
329 */
330 float feedforwardThrottle(float desiredSpeedKmh) const {
331 if (maxSpeedKmh <= 0.0f) return 0.0f;
332 return 100.0f * desiredSpeedKmh / maxSpeedKmh;
333 }

335 /**
336 * @brief Compute the throttle percent based on the selected mode.
337 */
338 float getThrottlePercent(float currentSpeedKmh) {
339 float speedError = desiredSpeedKmh - currentSpeedKmh;
340 float ff = feedforwardThrottle(desiredSpeedKmh);
341 float fb = pidSpeed_.calculate(0.0f, -speedError);
342 float throttlePercent = 0.0f;
343 switch (throttleMode) {
344 case ThrottleMode::FeedforwardOnly:
345 throttlePercent = ff;
346 break;
347 case ThrottleMode::PIDOnly:
348 throttlePercent = fb;
349 break;
350 case ThrottleMode::Combined:
351 default:
352 throttlePercent = ff + fb;
353 break;
354 }
355 // Clamp throttle to [0, 100]
356 if (throttlePercent > 100.0f) throttlePercent = 100.0f;
357 if (throttlePercent < 0.0f) throttlePercent = 0.0f;
358 if (throttlePercent > 0.0f && throttlePercent < 1.0f)
359 throttlePercent = 1.0f; // avoid too low throttle
360 return throttlePercent;
361 }

363 void sendControlMessages(float distanceToTarget, float headingError,
364 float currentSpeedKmh, float desiredHeading,
365 float currentHeading) {
366 resultThrottlePercent = getThrottlePercent(currentSpeedKmh);
367 Message<float> msg;
368 msg.source = MessageOrigin::Autonomy;
369 msg.content = MessageContent::Throttle;
370 msg.unit = Unit::Percent;
371 msg.value = resultThrottlePercent;
372 sendMessage(msg);
373 msg.content = MessageContent::SteeringAngle;
374 msg.unit = Unit::AngleDegree;
375 // ROS convention: positive steering = left turn. Invert heading error sign
376 // so negative error (target left) yields negative steering (right turn).
377 float pidInput =
378 !isInverted ? -headingError : headingError; // Invert error if needed
379 msg.value = resultSteeringAngleDeg = pidSteering_.calculate(0.0f, pidInput);
380 sendMessage(msg);
381 TRLogger.info(
382 "MotionController2D: desiredHeading=%.1f deg, currentHeading=%.1f deg, "
383 "headingError=%.1f deg, steeringAngle=%.1f deg, distanceToTarget=%.2f "
384 "m, desiredSpeed=%.2f km/h, currentSpeed=%.2f km/h, throttle=%.1f%%",
385 desiredHeading, currentHeading, headingError, resultSteeringAngleDeg,
386 distanceToTarget, desiredSpeedKmh, currentSpeedKmh,
387 resultThrottlePercent);
388 }

390 /// Handles dt initialization from first 10 updates, but does not block
391 /// control logic
392 bool initializeDtFromUpdates() {
393 unsigned long nowMs = millis();
394 if (dtSetFromUpdates) return true;
395 if (updateCount == 0) {
396 updateStartTimeMs = nowMs;
397 }
398 updateCount++;
399 if (updateCount == 11) {
400 updateEndTimeMs = nowMs;
401 float avgMs = (updateEndTimeMs - updateStartTimeMs) / 10.0f;
402 float dt = avgMs / 1000.0f; // convert ms to seconds
403 // Reconfigure both PIDs with new dt, keep other params
404 pidSpeed_.setDt(dt);
405 pidSteering_.setDt(dt);
406 dtSetFromUpdates = true;
407 }
408 return dtSetFromUpdates;
409 }
