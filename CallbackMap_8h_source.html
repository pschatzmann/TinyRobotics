11 * @class CallbackMap
12 * @brief A map-like utility for generating and validating neighbor coordinates
13 * in a configurable pattern.
14 *
15 * CallbackMap generates neighbor coordinates around a given point at a
16 * specified distance and angular resolution. It supports a user-provided
17 * callback to determine if a coordinate is valid (e.g., obstacle-free).
18 *
19 * Typical use: pathfinding, navigation, or virtual occupancy grids where
20 * neighbors are not stored but computed on demand.
21 *
22 * @tparam T Numeric type for coordinates (default: DistanceM)
23 *
24 * Example usage:
25 * @code
26 * CallbackMap<> map(1.0f, 16); // 1m distance, 16 directions
27 * map.setIsValidCallback([](Coordinate<DistanceM> c) { return c.x > 0; });
28 * auto neighbors = map.getNeighbors(Coordinate<DistanceM>(0,0));
29 * @endcode
30 */
31

33class CallbackMap {

43 return true; // Default to all cells valid if no callback provided

50 /// Get world coordinates of neighboring cells (for pathfinding or navigation)
51 std::vector<Coordinate<T>> getNeighbors(Coordinate<T> from) const {
52 std::vector<Coordinate<T>> neighbors;
53 // Generate neighbors at defaultDistance in defaultResolution directions
54 for (int i = 0; i < defaultNeighborCount; ++i) {
55 float angle = 2.0f * M_PI * i / defaultNeighborCount;
56 float dx = defaultDistance * std::cos(angle);
57 float dy = defaultDistance * std::sin(angle);
58 Coordinate<T> neighbor(from.x + dx, from.y + dy);
59 if (isValid(neighbor)) {
60 neighbors.push_back(neighbor);
61 }
62 }
63 return neighbors;
64 }
65
66 void setDistance(float distance) { defaultDistance = distance; }
67 void setNeighborCount(int neighborCount) {
68 defaultNeighborCount = neighborCount;
69 }
70
71 protected:
72 int defaultNeighborCount =
73 36; // Default to 36 neighbors (10 degree resolution)
74 float defaultDistance = 1.0f;
75 bool (*isValidCB)(Coordinate<T> coord) =
76 nullptr; // Optional callback for cell validity
