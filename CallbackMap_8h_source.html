11/**
12 * @class CallbackMap
13 * @brief A map-like utility for generating and validating neighbor coordinates
14 * in a configurable pattern.
15 *
16 * CallbackMap generates neighbor coordinates around a given point at a
17 * specified distance and angular resolution. It supports a user-provided
18 * callback to determine if a coordinate is valid (e.g., obstacle-free).
19 *
20 * Typical use: pathfinding, navigation, or virtual occupancy grids where
21 * neighbors are not stored but computed on demand.
22 *
23 * @tparam T Numeric type for coordinates (default: DistanceM)
24 *
25 * Example usage:
26 * @code
27 * CallbackMap<> map(1.0f, 16); // 1m distance, 16 directions
28 * map.setIsValidCallback([](Coordinate<DistanceM> c) { return c.x > 0; });
29 * auto neighbors = map.getNeighbors(Coordinate<DistanceM>(0,0));
30 * @endcode
31 */
32
33template <typename T = DistanceM>
34class CallbackMap : public IMap<T> {
35 public:

38 CallbackMap() = default;

40 /// @brief Construct a CallbackMap with extents, resolution, and neighbor count.
41 /// @param distanceX Map extent in X direction (meters)
42 /// @param distanceY Map extent in Y direction (meters)
43 /// @param resolutionM Grid resolution (meters)
44 /// @param neighborCount Number of neighbors to generate
45 CallbackMap(float distanceX, float distanceY, float resolutionM, int neighborCount)

51 /// @brief Get the number of cells in the X direction.
52 int getXCount() const override {
53 return (defaultResolutionM > 0 && distanceX > 0) ? static_cast<int>(distanceX / defaultResolutionM) : 0;
54 }
55
56 /// @brief Get the number of cells in the Y direction.
57 int getYCount() const override {
58 return (defaultResolutionM > 0 && distanceY > 0) ? static_cast<int>(distanceY / defaultResolutionM) : 0;
59 }
60
61 /// @brief Get the map resolution in meters.
62 float getResolution() const override { return defaultResolutionM; }
63
64
65 /// @brief Generate valid neighbor coordinates around a given point.
66 /// @param from Center coordinate
67 /// @return Vector of valid neighbor coordinates
68 std::vector<Coordinate<T>> getNeighbors(Coordinate<T> from) const override {
69 std::vector<Coordinate<T>> neighbors;
70 for (int i = 0; i < defaultNeighborCount; ++i) {
71 float angle = 2.0f * M_PI * i / defaultNeighborCount;
72 float dx = defaultResolutionM * std::cos(angle);
73 float dy = defaultResolutionM * std::sin(angle);
74 Coordinate<T> neighbor(from.x + dx, from.y + dy);
75 if (isValid(neighbor)) {
76 neighbors.push_back(neighbor);
77 }
78 }
79 return neighbors;
80 }
81
82
83 /// @brief Check if a coordinate is valid (not occupied).
84 /// @param coord Coordinate to check
85 /// @return True if valid (not OCCUPIED), false otherwise
86 bool isValid(const Coordinate<T>& coord) const override {
87 if (isValidCB) {
88 return isValidCB(coord) != CellState::OCCUPIED;
89 }
90 // If no callback is set, treat all cells as valid
91 return true;
92 }
93
94
95 /// @brief Get the cell state at grid coordinates (x, y).
96 /// @param x X grid index
97 /// @param y Y grid index
98 /// @param state Output cell state
99 /// @return True if cell state is available
100 bool getCell(int x, int y, CellState& state) const override {
101 Coordinate<T> c = toWorld(x, y);
102 if (isValidCB) {
103 state = isValidCB(c);
104 return true;
105 }
106 return false;
107 }

113 /// @return World coordinate
114 Coordinate<T> toWorld(int x, int y) const override {

116 }

122 isValidCB = callback;

142
