13 * @class Distance
14 * @ingroup units
15 * @brief Represents a distance measurement with unit conversion support.
16 *
17 * The Distance class encapsulates a distance value and its unit, supporting meters (M),
18 * centimeters (CM), millimeters (MM), and feet (FEET). It provides methods to set and retrieve
19 * the distance in any supported unit, automatically handling conversions.
20 *
21 * - Internal state is always consistent with the last set value and unit.
22 * - Designed for embedded and robotics applications where unit flexibility and efficiency are required.
23 * - Use getValue() to retrieve the distance in any unit; use setValue() to update the value and unit.
24 *
25 * Example:
26 * @code
27 * Distance d(1.0, DistanceUnit::M);
28 * float feet = d.getValue(DistanceUnit::FEET); // Convert to feet
29 * d.setValue(100.0, DistanceUnit::CM); // Update value in centimeters
30 * float meters = d.getValue(DistanceUnit::M); // Convert back to meters
31 * @endcode
32 *
33 * @note Invalid conversions return -1.0f.
34 *
35 * @see DistanceUnit
36 */
37class Distance {
38 public:
39 Distance() = default;
40 Distance(float distance, DistanceUnit unit) { setValue(distance, unit); }
41
42 void setValue(float newDistance, DistanceUnit newUnit) {
43 distance = newDistance;
44 unit = newUnit;
45 }
46
47 float getValue(DistanceUnit desiredUnit) const {
48 if (unit == desiredUnit) return distance;
49 switch (unit) {
50 case DistanceUnit::M:
51 if (desiredUnit == DistanceUnit::CM) return distance * 100;
52 if (desiredUnit == DistanceUnit::MM) return distance * 1000;
53 if (desiredUnit == DistanceUnit::FEET) return distance * 3.28084;
54 break;
55 case DistanceUnit::CM:
56 if (desiredUnit == DistanceUnit::M) return distance / 100;
57 if (desiredUnit == DistanceUnit::MM) return distance * 10;
58 if (desiredUnit == DistanceUnit::FEET) return distance * 0.0328084;
59 break;
60 case DistanceUnit::MM:
61 if (desiredUnit == DistanceUnit::M) return distance / 1000;
62 if (desiredUnit == DistanceUnit::CM) return distance / 10;
63 if (desiredUnit == DistanceUnit::FEET) return distance * 0.00328084;
64 break;
65 case DistanceUnit::FEET:
66 if (desiredUnit == DistanceUnit::M) return distance / 3.28084;
67 if (desiredUnit == DistanceUnit::CM) return distance / 0.0328084;
68 if (desiredUnit == DistanceUnit::MM) return distance / 0.00328084;
69 break;
70 }
71 return -1; // Invalid conversion
72 }

74 protected:
