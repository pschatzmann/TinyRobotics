115 GPSCoordinate navigate(float distance_m, float bearing_deg,
116 float alt_diff_m = 0) const {
117 float R = 6371000; // Earth radius in meters
118 float bearing_rad = bearing_deg * M_PI / 180.0;
119 float lat1 = latitude * M_PI / 180.0;
120 float lon1 = longitude * M_PI / 180.0;
121
122 float lat2 = asin(sin(lat1) * cos(distance_m / R) +
123 cos(lat1) * sin(distance_m / R) * cos(bearing_rad));
124 float lon2 =
125 lon1 + atan2(sin(bearing_rad) * sin(distance_m / R) * cos(lat1),
126 cos(distance_m / R) - sin(lat1) * sin(lat2));
127
128 return GPSCoordinate(lat2 * 180.0 / M_PI, lon2 * 180.0 / M_PI,
129 altitude + alt_diff_m);
130 }

177 float bearingDegree(const GPSCoordinate& other) const {
178 float lat1 = latitude * M_PI / 180.0;
179 float lat2 = other.latitude * M_PI / 180.0;
180 float dLon = (other.longitude - longitude) * M_PI / 180.0;

195 float distanceM(const GPSCoordinate& other) const {
196 float R = 6371000; // Earth radius in meters
197 float lat1 = latitude * M_PI / 180.0;
198 float lat2 = other.latitude * M_PI / 180.0;
199 float dLat = (other.latitude - latitude) * M_PI / 180.0;
200 float dLon = (other.longitude - longitude) * M_PI / 180.0;
201
202 float a = sin(dLat / 2) * sin(dLat / 2) +
203 cos(lat1) * cos(lat2) * sin(dLon / 2) * sin(dLon / 2);
204 float c = 2 * atan2(sqrt(a), sqrt(1 - a));
205 return R * c; // Distance in meters
206 }
