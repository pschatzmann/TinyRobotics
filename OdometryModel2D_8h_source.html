10/**
11 * @class OdometryModelAckerman
12 * @ingroup odometry
13 * @brief Odometry model for Ackermann, differential drive, and boat kinematics.
14 *
15 * This model computes the change in heading (theta) for:
16 * - Ackermann steering vehicles (car-like, with nonzero wheelBase)
17 * - Differential drive robots (wheelBase = 0, steeringAngle is angular velocity)
18 * - Boats (set wheelBase to the distance to the rudder; steeringAngle is rudder angle)
19 *
20 * ## Kinematic Handling
21 * - If wheelBase > 0: Uses Ackermann/boat kinematics: omega = v * tan(steeringAngle) / wheelBase
22 * - If wheelBase == 0: Uses differential drive convention: omega = steeringAngle (interpreted as angular velocity)
23 *
24 * ## Boat Note
25 * For boats, set wheelBase to the distance from the center of mass to the rudder, and provide the rudder angle as steeringAngle.
26 *
27 * @author TinyRobotics contributors
28 * @date 2026-04-07
29 */

40 /**
41 * @brief Compute heading change (deltaTheta) for Ackermann kinematics.
42 * @param speed Speed of the vehicle (meters/second)
43 * @param steeringAngle Steering angle (radians)
44 * @param deltaTimeMs Time interval (milliseconds)
45 * @return Change in heading (radians)
46 */
47 float computeDeltaTheta(uint16_t deltaTimeMs) const override {
48 float wb = wheelBase.getValue(DistanceUnit::M);
49 float speedMps = speed.getValue(SpeedUnit::MPS);
50 float steeringAngleRad = steeringAngle.getValue(AngleUnit::RAD);
51 float omega = (wb > 0.0f) ? speedMps * std::tan(steeringAngleRad) / wb : steeringAngleRad;
52 return omega * static_cast<float>(deltaTimeMs) / 1000.0f;
53 }

56 float speedMps = speed.getValue(SpeedUnit::MPS);
57 float dt = deltaTimeMs / 1000.0f;
58 deltaX = speedMps * std::cos(theta) * dt;
59 deltaY = speedMps * std::sin(theta) * dt;
