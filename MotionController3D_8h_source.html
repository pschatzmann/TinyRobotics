117 bool update() {
118 if (!is_active) return false;
119 advanceWaypoint(positionToleranceM);
120 if (path.isEmpty()) {
121 return false;
122 }
123 // Use the referenced motionState for current state
124 float vx =
125 pidX.calculate(target.getPosition().x, motionState.getPosition().x);
126 float vy =
127 pidY.calculate(target.getPosition().y, motionState.getPosition().y);
128 float vz =
129 pidZ.calculate(target.getPosition().z, motionState.getPosition().z);
130 float vyaw = pidYaw.calculate(wrapAngle(target.getOrientation().yaw),
131 wrapAngle(motionState.getOrientation().yaw));
132 float vpitch =
133 pidPitch.calculate(wrapAngle(target.getOrientation().pitch),
134 wrapAngle(motionState.getOrientation().pitch));
135 float vroll =
136 pidRoll.calculate(wrapAngle(target.getOrientation().roll),
137 wrapAngle(motionState.getOrientation().roll));

139 linearCmd = Speed3D(vx, vy, vz, SpeedUnit::MPS);
140 angularCmd = AngularVelocity3D(vyaw, vpitch, vroll,

205 bool advanceWaypoint(float positionTolerance = 0.1f) {
206 if (path.isEmpty()) return false;
207 auto tgt = path[0];
208 float dx = tgt.x - motionState.getPosition().x;
209 float dy = tgt.y - motionState.getPosition().y;
210 float dz = tgt.z - motionState.getPosition().z;
211 float dist = sqrtf(dx * dx + dy * dy + dz * dz);
212 if (dist < positionTolerance) {
213 path.removeHead();
214 if (!path.isEmpty()) {
215 // Set next waypoint as new target (keep orientation/vel from previous
216 // target)
217 if (path.size() == 1) {
218 // hover at coordinate
219 target = MotionState3D(
220 path[0], target.getOrientation(),
221 Speed3D(0, 0, 0, SpeedUnit::MPS),
222 AngularVelocity3D(0, 0, 0, AngularVelocityUnit::RadPerSecond));
223 } else {
224 target =
225 MotionState3D(path[0], target.getOrientation(), target.getSpeed(),
226 target.getAngularVelocity());
227 }
228 } else {
229 // Path is now empty, trigger on-goal action
230 handleOnGoalAction();
231 }
232 return true;
233 }
234 return false;
235 }
236 // For Circle mode
237 float circlePhase = 0.0f;
238 float circleRadius = 5.0f; // meters
239 float circleAngularSpeed = 0.5f; // radians per update
240 Coordinate<DistanceM> circleCenter;
241 bool circleInitialized = false;
242
243 void handleOnGoalAction() {
244 // process callback
245 if (onGoalCallback) {
246 if (onGoalCallback(onGoaldRef)) {
247 // Callback handled the goal action, so we can return early
248 return;
249 }
250 }
251 switch (onGoalAction) {
252 case OnGoalAction::Stop:
253 // Set all commands to zero and deactivate controller
254 linearCmd = Speed3D(0, 0, 0, SpeedUnit::MPS);
255 angularCmd =
256 AngularVelocity3D(0, 0, 0, AngularVelocityUnit::RadPerSecond);
257 is_active = false;
258 break;
259 case OnGoalAction::HoldPosition:
260 // Keep controller active, hold at last target (do nothing)
261 break;
262 case OnGoalAction::Circle:
263 // Move in a circle around the last goal
264 if (!circleInitialized) {
265 circleCenter = motionState.getPosition();
266 circlePhase = 0.0f;
267 circleInitialized = true;
268 } else {
269 circlePhase += circleAngularSpeed;
270 if (circlePhase > 2 * M_PI) circlePhase -= 2 * M_PI;
271 }
272 // Compute new target on the circle (XY plane, keep Z constant)
273 float x = circleCenter.x + circleRadius * cosf(circlePhase);
274 float y = circleCenter.y + circleRadius * sinf(circlePhase);
275 float z = circleCenter.z;
276 // Optionally, face tangent to the circle (yaw)
277 float yaw = atan2f(y - circleCenter.y, x - circleCenter.x) + M_PI_2;
278 Orientation3D orientation(yaw, target.getOrientation().pitch,
279 target.getOrientation().roll);
280 target = MotionState3D(
281 Coordinate<DistanceM>(x, y, z), orientation,
282 Speed3D(0, 0, 0, SpeedUnit::MPS),
283 AngularVelocity3D(0, 0, 0, AngularVelocityUnit::RadPerSecond));
284 break;
285 }
286 }

288 static float wrapAngle(float a) {
289 while (a > M_PI) a -= 2 * M_PI;
290 while (a < -M_PI) a += 2 * M_PI;
291 return a;
292 }
