17/**
18 * @class MotionController3D
19 * @ingroup control
20 * @brief 3D motion/path controller using PID for position and orientation.
21 *
22 * This class provides high-level 3D path following and pose control for robots,
23 * drones, or vehicles. It uses independent PID controllers for each axis (x, y,
24 * z) and orientation (yaw, pitch, roll).
25 *
26 * ## Features
27 * - Accepts a reference to an IMotionState3D for real-time feedback
28 * - Supports a path of 3D waypoints (Path<Coordinate<DistanceM>>)
29 * - Automatically advances to the next waypoint when the current one is reached
30 * - PID gains and limits are configurable for both position and orientation
31 * - Provides update() to compute new commands, and getters for linear/angular
32 * velocity commands
33 * - begin() initializes the target from the first path coordinate
34 *
35 * ## Usage
36 * 1. Create with a reference to your IMotionState3D implementation (e.g.,
37 * IMU3D)
38 * 2. Set a path of waypoints with setPath()
39 * 3. Call begin() to initialize the first target
40 * 4. In your control loop:
41 * - Call update() to compute new commands
42 * - Use getLinearCommand() and getAngularCommand() for actuation
43 *
44 * ## Limitations
45 * - Only the position is updated from the path; orientation, speed, and angular
46 * velocity are kept from the previous target
47 *
48 * ## Example
49 * @code
50 * MotionController3D controller(imu3d);
51 * controller.setPath(path);
52 * controller.begin();
53 * while (!controller.isPathComplete()) {
54 * controller.update();
55 * auto v = controller.getLinearCommand();
56 * auto w = controller.getAngularCommand();
57 * // send v, w to actuators
58 * controller.advanceWaypoint();
59 * }
60 * @endcode
61 */

219 bool initializeDtFromUpdates() {
220 unsigned long nowMs = millis();

242 bool advanceWaypoint(float positionTolerance = 0.1f) {
243 if (path.isEmpty()) return false;
244 auto tgt = path[0];
245 float dx = tgt.x - motionState.getPosition().x;
246 float dy = tgt.y - motionState.getPosition().y;
247 float dz = tgt.z - motionState.getPosition().z;
248 float dist = std::sqrt(dx * dx + dy * dy + dz * dz);
249 if (dist < positionTolerance) {
250 path.removeHead();
251 if (!path.isEmpty()) {
252 // Set next waypoint as new target (keep orientation/vel from previous
253 // target)
254 if (path.size() == 1) {
255 // hover at coordinate
256 target = MotionState3D(
257 path[0], target.getOrientation(),
258 Speed3D(0, 0, 0, SpeedUnit::MPS),
259 AngularVelocity3D(0, 0, 0, AngularVelocityUnit::RadPerSec));
260 } else {
261 target =
262 MotionState3D(path[0], target.getOrientation(), target.getSpeed(),
263 target.getAngularVelocity());
264 }
265 } else {
266 // Path is now empty, trigger on-goal action
267 handleOnGoalAction();
268 }
269 return true;
270 }
271 return false;
272 }
273 // For Circle mode
274 float circlePhase = 0.0f;
275 float circleRadius = 5.0f; // meters
276 float circleAngularSpeed = 0.5f; // radians per update
277 Coordinate<DistanceM> circleCenter;
278 bool circleInitialized = false;
279
280 void handleOnGoalAction() {
281 // process callback
282 if (onGoalCallback) {
283 if (onGoalCallback(onGoaldRef)) {
284 // Callback handled the goal action, so we can return early
285 return;
286 }
287 }
288 switch (onGoalAction) {
289 case OnGoalAction::Stop:
290 // Set all commands to zero and deactivate controller
291 linearCmd = Speed3D(0, 0, 0, SpeedUnit::MPS);
292 angularCmd = AngularVelocity3D(0, 0, 0, AngularVelocityUnit::RadPerSec);
293 is_active = false;
294 break;
295 case OnGoalAction::HoldPosition:
296 // Keep controller active, hold at last target (do nothing)
297 break;
298 case OnGoalAction::Circle:
299 // Move in a circle around the last goal
300 if (!circleInitialized) {
301 circleCenter = motionState.getPosition();
302 circlePhase = 0.0f;
303 circleInitialized = true;
304 } else {
305 circlePhase += circleAngularSpeed;
306 if (circlePhase > 2.0f * static_cast<float>(M_PI))
307 circlePhase -= 2.0f * static_cast<float>(M_PI);
308 }
309 // Compute new target on the circle (XY plane, keep Z constant)
310 float x = circleCenter.x + circleRadius * std::cos(circlePhase);
311 float y = circleCenter.y + circleRadius * std::sin(circlePhase);
312 float z = circleCenter.z;
313 // Optionally, face tangent to the circle (yaw)
314 float yaw = std::atan2(y - circleCenter.y, x - circleCenter.x) +
315 static_cast<float>(M_PI_2);
316 Orientation3D orientation(yaw, target.getOrientation().pitch,
317 target.getOrientation().roll);
318 target = MotionState3D(
319 Coordinate<DistanceM>(x, y, z), orientation,
320 Speed3D(0, 0, 0, SpeedUnit::MPS),
321 AngularVelocity3D(0, 0, 0, AngularVelocityUnit::RadPerSec));
322 break;
323 }
324 }
