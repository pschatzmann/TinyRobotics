96 // 3. Decide type and read value accordingly

99 // Read Coordinate<float> value

105 result.coordMsg.source_id = source_id;
106 result.coordMsg.content = content;
107 result.coordMsg.unit = unit;

109 io.readBytes((char*)&result.coordMsg.source, sizeof(MessageOrigin));
110 } else if (content == MessageContent::PositionGPS) {
111 result.type = MessageValueType::GPSCoordinate;
