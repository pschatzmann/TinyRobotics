16/**
17 * @class Fusion2D
18 * @ingroup fusion
19 * @brief 2D sensor fusion EKF for position, heading, and speed estimation.
20 *
21 * Supports optional sensors:
22 * - IMU (gyro + optional acceleration)
23 * - Wheel speed
24 * - Absolute heading (magnetometer or GPS-derived)
25 * - GPS position
26 *
27 * The EKF automatically adapts to the available sensors.
28 */

68 // =====================
69 // PREDICTION
70 // =====================
71 /**
72 * @brief Predict the next state based on elapsed time.
73 *
74 * This method performs dead-reckoning using the last speed and heading.
75 * If no wheel speed is available but IMU acceleration is provided, it will
76 * integrate ax/ay to estimate speed.
77 *
78 * @param timeMs Current timestamp in milliseconds.
79 * @param ax Optional IMU acceleration along body-forward axis (m/s²). Used
80 * only if no wheel speed.
81 * @param ay Optional IMU acceleration along body-sideways axis (m/s²). Used
82 * only if no wheel speed.
83 */
84 void predict(uint32_t timeMs, float ax = 0.0f, float ay = 0.0f) {
85 uint32_t dtMs = timeMs - lastPredictTime;
86 lastPredictTime = timeMs;
87 float dt = dtMs * 0.001f;
88 if (dt <= 0.0f) return;
89 if (dt > 0.1f) dt = 0.1f;
90
91 float c = cosf(state.heading);
92 float s = sinf(state.heading);
93
94 // Determine velocity
95 float v = state.speed;
96
97 // Use IMU acceleration if wheel speed not available
98 if (!hasWheelSpeed && hasIMU) {
99 float ax_world = ax * c - ay * s;
100 float ay_world = ax * s + ay * c;
101 float dv = sqrtf(ax_world * ax_world + ay_world * ay_world) * dt;
102 v += dv;
103 state.speed = v;
104 }
105
106 // Heading integration
107 float omega = hasIMU ? (omegaMeasured - state.gyroBias) : 0.0f;
108 state.heading += omega * dt;
109 state.heading = wrapAngle(state.heading);
110
111 // Position integration (requires some velocity)
112 if (v > 0.0f) {
113 state.x += v * c * dt;
114 state.y += v * s * dt;
115 }
116
117 // EKF covariance propagation
118 float F[5][5] = {{1, 0, -v * s * dt, c * dt, 0},
119 {0, 1, v * c * dt, s * dt, 0},
120 {0, 0, 1, 0, -dt},
121 {0, 0, 0, 1, 0},
122 {0, 0, 0, 0, 1}};
123
124 float q_pos = 0.05f, q_heading = 0.01f, q_speed = 0.1f, q_bias = 0.0005f;
125 float Q[5][5] = {{q_pos, 0, 0, 0, 0},
126 {0, q_pos, 0, 0, 0},
127 {0, 0, q_heading, 0, 0},
128 {0, 0, 0, q_speed, 0},
129 {0, 0, 0, 0, q_bias}};
130
131 float FP[5][5] = {}, FPFt[5][5] = {};
132 for (int i = 0; i < 5; i++)
133 for (int j = 0; j < 5; j++)
134 for (int k = 0; k < 5; k++) FP[i][j] += F[i][k] * P[k][j];
135 for (int i = 0; i < 5; i++)
136 for (int j = 0; j < 5; j++)
137 for (int k = 0; k < 5; k++) FPFt[i][j] += FP[i][k] * F[j][k];
138 for (int i = 0; i < 5; i++)
139 for (int j = 0; j < 5; j++) P[i][j] = FPFt[i][j] + Q[i][j];
140
141 state.timeMs = timeMs;
142 }

147 /**
148 * @brief Update the EKF with IMU measurements.
149 *
150 * @param timeMs Timestamp of the IMU measurement.
151 * @param yawRate Gyro z-axis measurement (yaw rate, rad/s).
152 * @param ax Optional IMU acceleration along body x-axis (m/s²).
153 * @param ay Optional IMU acceleration along body y-axis (m/s²).
154 *
155 * The yawRate is used for heading prediction, and optional ax/ay
156 * are used to integrate speed when wheel encoder is missing.
157 */
158 void updateIMU(uint32_t timeMs, float yawRate, float ax = 0.0f,
159 float ay = 0.0f) {
160 hasIMU = true;
161 omegaMeasured = yawRate;
162 predict(timeMs, ax, ay);
163 }
164
165 // Update filter using an IMU2D instance
166 void updateIMU(const IMU2D<float>& imu) {
167 uint32_t t = imu.getTimestamp(); // timestamp in ms or µs
168 float omega = imu.getAngularVelocity(); // angular velocity (rad/s)
169 float v = imu.getLinearVelocity(); // linear velocity (m/s)
170
171 // 1. Integrate angular velocity into orientation
172 updateIMU(t, omega);
173
174 // 2. Integrate linear velocity into position/speed
175 updateSpeed(t, v);
176 }
177
178 /**
179 * @brief Update the EKF with wheel speed.
180 *
181 * @param timeMs Timestamp of the speed measurement.
182 * @param speed Wheel speed in meters per second.
183 *
184 * Overrides IMU acceleration-based speed if available.
185 */
186 void updateSpeed(uint32_t timeMs, float speed) {
187 hasWheelSpeed = true;
188 predict(timeMs);
189 float y = speed - state.speed;
190 float R = 0.5f;
191 float S = P[3][3] + R;
192 float K[5];
193 for (int i = 0; i < 5; i++) K[i] = P[i][3] / S;
194 applyUpdate(K, y);
195 updateCovariance(K, 3);
196 }
197
198 void updateSpeed(uint32_t timeMs, Speed newSpeed) {
199 updateSpeed(timeMs, newSpeed.getValue(SpeedUnit::MPS));
200 }
201
202 /**
203 * @brief Update the EKF with an absolute heading measurement.
204 *
205 * @param timeMs Timestamp of the heading measurement.
206 * @param heading Absolute heading in radians (e.g., from magnetometer or
207 * GPS-derived).
208 *
209 * This corrects the heading estimate and helps reduce gyro drift.
210 */
211 void updateHeading(uint32_t timeMs, float heading) {
212 predict(timeMs);
213 float y = wrapAngle(heading - state.heading);
214 float R = 0.3f;
215 float S = P[2][2] + R;
216 float K[5];
217 for (int i = 0; i < 5; i++) K[i] = P[i][2] / S;
218 applyUpdate(K, y);
219 state.heading = wrapAngle(state.heading);
220 updateCovariance(K, 2);
221 }
222
223 void updateHeading(uint32_t timeMs, Angle newHeading) {
224 updateHeading(timeMs, newHeading.getValue(AngleUnit::RAD));
225 }
226
227 /**
228 * @brief Update the EKF with GPS position.
229 *
230 * @param timeMs Timestamp of the GPS measurement.
231 * @param x GPS X position in meters (or local coordinates).
232 * @param y GPS Y position in meters (or local coordinates).
233 * @param accuracy Standard deviation of the GPS measurement (meters).
234 *
235 * Corrects the EKF position estimate. Works independently of IMU or wheel
236 * speed.
237 */
238 void updateGPS(uint32_t timeMs, float x, float y, float accuracy) {
239 predict(timeMs);
240 float R = accuracy * accuracy;
241 // X
242 {
243 float residual = x - state.x;
244 float S = P[0][0] + R;
245 float K[5];
246 for (int i = 0; i < 5; i++) K[i] = P[i][0] / S;
247 applyUpdate(K, residual);
248 updateCovariance(K, 0);
249 }
250 // Y
251 {
252 float residual = y - state.y;
253 float S = P[1][1] + R;
254 float K[5];
255 for (int i = 0; i < 5; i++) K[i] = P[i][1] / S;
256 applyUpdate(K, residual);
257 updateCovariance(K, 1);
258 }
259 }
260
261 void updateGPS(uint32_t timeMs, const Coordinate<DistanceM>& gpsCoord,
262 float accuracyM) {
263 updateGPS(timeMs, gpsCoord.x, gpsCoord.y, accuracyM);
264 }

299 private:
300 State2D state;
301 float P[5][5] = {};
302 uint32_t lastPredictTime = 0;
303 float omegaMeasured = 0.0f;
304 bool hasWheelSpeed = false;
305 bool hasIMU = false;
306
307 void applyUpdate(float K[5], float residual) {
308 state.x += K[0] * residual;
309 state.y += K[1] * residual;
310 state.heading += K[2] * residual;
311 state.speed += K[3] * residual;
312 state.gyroBias += K[4] * residual;
313 state.heading = wrapAngle(state.heading);
314 }
315
316 void updateCovariance(float K[5], int idx) {
317 float newP[5][5];
318 for (int i = 0; i < 5; i++)
319 for (int j = 0; j < 5; j++) newP[i][j] = P[i][j] - K[i] * P[idx][j];
320 for (int i = 0; i < 5; i++)
321 for (int j = 0; j < 5; j++) P[i][j] = newP[i][j];
322 }
323
324 static float wrapAngle(float a) {
325 while (a > M_PI) a -= 2 * M_PI;
326 while (a < -M_PI) a += 2 * M_PI;
327 return a;
328 }

330 /**
331 * @brief Reset the EKF state and covariance.
332 * @param timeMs Optional timestamp to set the initial state time.
333 */
334 void reset(uint32_t timeMs = millis()) {
335 state = {};
336 state.timeMs = timeMs;
337 state.speed = 0.0f;
338 state.heading = 0.0f;
339 state.gyroBias = 0.0f;
340
341 hasWheelSpeed = false;
342 hasIMU = false;
343
344 for (int i = 0; i < 5; i++)
345 for (int j = 0; j < 5; j++) P[i][j] = (i == j) ? 1.0f : 0.0f;
346
347 lastPredictTime = timeMs;
348 omegaMeasured = 0.0f;
349 }
