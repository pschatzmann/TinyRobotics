82 // This is now private; use static trampoline for MQTT callback
83 void onMqttMessage(char* topic, uint8_t* payload, unsigned int length) {
84 std::string t(topic);
85 if (endsWith(t, baseTopic_ + "value") && length == sizeof(Message<float>)) {
86 Message<float> msg;
87 memcpy(&msg, payload, sizeof(msg));
88 handler_.onMessage(msg);
89 } else if (endsWith(t, baseTopic_ + "coordinate") &&
90 length == sizeof(Message<Coordinate<float>>)) {
91 Message<Coordinate<float>> msg;
92 memcpy(&msg, payload, sizeof(msg));
93 handler_.onMessage(msg);
94 } else if (endsWith(t, baseTopic_ + "gps") &&
95 length == sizeof(Message<GPSCoordinate>)) {
96 Message<GPSCoordinate> msg;
97 memcpy(&msg, payload, sizeof(msg));
98 handler_.onMessage(msg);
99 } else if (endsWith(t, baseTopic_ + "motionstate") &&
100 length == sizeof(Message<MotionState3D>)) {
101 Message<MotionState3D> msg;
102 memcpy(&msg, payload, sizeof(msg));
103 handler_.onMessage(msg);
104 }
105 }
106
107 // Static trampoline for PubSubClient callback
108 static void mqttCallbackTrampoline(char* topic, uint8_t* payload,
109 unsigned int length) {
110 if (instance_) instance_->onMqttMessage(topic, payload, length);
111 }
