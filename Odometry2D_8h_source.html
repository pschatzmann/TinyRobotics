12/**
13 * @class Odometry2D
14 * @ingroup odometry
15 * @brief Tracks 2D position and orientation of a robot using velocity and
16 * steering data.
17 *
18 * This class provides simple 2D odometry for mobile robots, such as
19 * differential drive or Ackermann vehicles. It integrates velocity and steering
20 * angle over time to estimate the robot's position (x, y) in meters.
21 *
22 * ## Supported Kinematics
23 * - Differential drive (default model)
24 * - Ackermann steering (if steering angle is provided)
25 *
26 * ## Inputs
27 * - Speed (as a Speed object, e.g., meters per second)
28 * - Steering angle (as an Angle object, e.g., radians)
29 * - Time delta (in milliseconds, or uses millis() if not provided)
30 *
31 * ## Outputs
32 * - 2D position (x, y) in meters
33 *
34 * ## Coordinate Frame
35 * - All positions are in the robot's local/world frame, with x forward and y
36 * left/right.
37 *
38 * ## Update Method
39 * - Call update() with new speed and steering angle at each control loop
40 * iteration.
41 * - Optionally provide delta time, or let the class compute it using millis().
42 *
43 * ## Integration Method
44 * - Uses simple Euler integration for position updates.
45 *
46 * ## Limitations
47 * - Assumes no wheel slip or drift.
48 * - Not suitable for holonomic or omnidirectional robots.
49 * - Orientation (theta) is not explicitly tracked in this class.
50 *
51 * ## Example Usage
52 * @code
53 * Odometry2D odom;
54 * odom.begin(Coordinate<DistanceM>(0, 0));
55 * // In your control loop:
56 * odom.update(currentSpeed, currentSteeringAngle);
57 * auto pos = odom.getPosition();
58 * Serial.printf("x=%.2f, y=%.2f\n", pos.x, pos.y);
59 * @endcode
60 *
61 * @author TinyRobotics contributors
62 * @date 2026-03-30
63 */

65class Odometry2D : public MessageSource, public IMotionState2D {
66
67 public:
68 Odometry2D() = default;

70 /**
71 * @brief Initialize the odometry state.
72 *
73 * @param initialPosition The starting position of the robot (x, y) in meters.
74 * @param initialTheta The starting orientation (heading/yaw) in radians.
75 * Default is 0.0.
76 * @param wheelBase The wheelbase (distance between axles for Ackermann, or
77 * length to rudder for boats) in meters. Default is 0 (differential drive).
78 * @return true on success
79 *
80 * If wheelBase is set (>0), Ackermann or boat kinematics are used for heading
81 * updates. If wheelBase is zero, differential drive kinematics are used.
82 */
83 bool begin(Coordinate<DistanceM> initialPosition, float initialTheta = 0.0f,
84 Distance wheelBase = Distance()) {
85 position = initialPosition;
86 theta = initialTheta;
87 totalDistance = 0.0f;
88 lastUpdateTimeMs = 0;
89 this->wheelBase = wheelBase;
90 return true;
91 }
92
93 /**
94 * @brief Initialize the odometry state from a Frame2D.
95 *
96 * @param frame The Frame2D containing the initial position and orientation.
97 * @param wheelBase The wheelbase (optional, default 0).
98 * @return true on success
99 */
100 bool begin(const Frame2D& frame, Distance wheelBase = Distance()) {
101 // Extract position and orientation from the frame's transform
102 auto tf = frame.getTransform();
103 // tf.translation is a Coordinate<float>, convert to Coordinate<DistanceM>
104 return begin(tf.pos, tf.getHeading(AngleUnit::RAD), wheelBase);
105 }
106
107 /**
108 * @brief Update the odometry state with new speed and steering angle.
109 *
110 * @param speed The current speed of the robot (with units).
111 * @param steeringAngle The current steering angle (radians for
112 * Ackermann/rudder, or angular velocity for differential drive).
113 * @param deltaTimeMs Time since last update in milliseconds.
114 *
115 * If wheelBase is set (>0), uses Ackermann/boat kinematics for heading
116 * update: omega = v * tan(steeringAngle) / wheelBase If wheelBase is zero,
117 * uses differential drive kinematics: deltaTheta = steeringAngle (angular
118 * velocity) * deltaTime
119 */
120 void update(Speed speed, Angle steeringAngle, float deltaTimeMs) {
121 this->steeringAngle = steeringAngle;
122 this->speed = speed;
123 float speedMps = speed.getValue(SpeedUnit::MPS);
124 float steeringAngleRad = steeringAngle.getValue(AngleUnit::RAD);
125 float deltaTheta = 0.0f;
126 // Use Ackermann if wheelBase is set, else differential drive
127 if (wheelBase.getValue(DistanceUnit::M) > 0.0f) {
128 float wb = wheelBase.getValue(DistanceUnit::M);
129 float omega = (wb > 0.0f) ? speedMps * tan(steeringAngleRad) / wb : 0.0f;
130 deltaTheta = omega * deltaTimeMs / 1000.0f;
131 } else {
132 // Differential drive: steeringAngleRad is angular velocity
133 deltaTheta = steeringAngleRad * deltaTimeMs / 1000.0f;
134 }
135 theta += deltaTheta;
136 float deltaX = speedMps * cos(theta) * deltaTimeMs / 1000.0f;
137 float deltaY = speedMps * sin(theta) * deltaTimeMs / 1000.0f;
138 position.x += deltaX;
139 position.y += deltaY;
140 lastDelta = {deltaX, deltaY, deltaTheta};
141 totalDistance += sqrt(deltaX * deltaX + deltaY * deltaY);
142
143 publish();
144 }
145
146 /**
147 * @brief Update the odometry state with new speed and steering angle, using
148 * automatic time delta.
149 *
150 * @param speed The current speed of the robot (with units).
151 * @param steeringAngle The current steering angle (radians for
152 * Ackermann/rudder, or angular velocity for differential drive).
153 *
154 * Uses millis() to compute the time since the last update.
155 */
156 void update(Speed speed, Angle steeringAngle) {
157 auto now = millis();
158 float deltaTimeMs = now - lastUpdateTimeMs;
159 if (lastUpdateTimeMs > 0) update(speed, steeringAngle, deltaTimeMs);
160 lastUpdateTimeMs = now;
161 }

190 }
