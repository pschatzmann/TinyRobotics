105 void update(Speed speed, Angle steeringAngle, float deltaTimeMs) {
106 this->steeringAngle = steeringAngle;
107 this->speed = speed;
108 float speedMps = speed.getValue(SpeedUnit::MPS);
109 float steeringAngleRad = steeringAngle.getValue(AngleUnit::RAD);
110 float deltaTheta = 0.0f;
111 // Use Ackermann if wheelBase is set, else differential drive
112 if (wheelBase.getValue(DistanceUnit::M) > 0.0f) {
113 float wb = wheelBase.getValue(DistanceUnit::M);
114 float omega = (wb > 0.0f) ? speedMps * tan(steeringAngleRad) / wb : 0.0f;
115 deltaTheta = omega * deltaTimeMs / 1000.0f;
116 } else {
117 // Differential drive: steeringAngleRad is angular velocity
118 deltaTheta = steeringAngleRad * deltaTimeMs / 1000.0f;
119 }
120 theta += deltaTheta;
121 float deltaX = speedMps * cos(theta) * deltaTimeMs / 1000.0f;
122 float deltaY = speedMps * sin(theta) * deltaTimeMs / 1000.0f;
123 position.x += deltaX;
124 position.y += deltaY;
125 lastDelta = {deltaX, deltaY, deltaTheta};
126 totalDistance += sqrt(deltaX * deltaX + deltaY * deltaY);
127
128 publish();
129 }
