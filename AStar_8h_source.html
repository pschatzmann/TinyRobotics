126 std::priority_queue<NodeRecord, std::vector<NodeRecord>,

129 std::unordered_map<Node, float, std::hash<Node>, std::equal_to<Node>, AllocatorPSRAM<std::pair<const Node, float>>> costSoFar;
130 openSet.push({start, 0.0f, cost_cb(start, goal, ref)});
131 costSoFar[start] = 0.0f;
132 if (cameFrom) cameFrom->clear();
133 while (!openSet.empty()) {
134 Node current = openSet.top().node;
135 if (current == goal) {
136 if (outCostSoFar) *outCostSoFar = costSoFar;
137 return true;
138 }
139 openSet.pop();
140 for (const auto& neighbor : map.getNeighbors(current)) {
141 if (!valid_cb(neighbor, ref)) continue;
142 float newCost = costSoFar[current] + cost_cb(current, neighbor, ref);
143 if (!costSoFar.count(neighbor) || newCost < costSoFar[neighbor]) {
144 costSoFar[neighbor] = newCost;
145 float priority = newCost + cost_cb(neighbor, goal, ref);
146 openSet.push({neighbor, newCost, priority});
147 if (cameFrom) (*cameFrom)[neighbor] = current;
148 }
149 }
150 }
151 if (outCostSoFar) *outCostSoFar = costSoFar;
152 return false;
153 }
